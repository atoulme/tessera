002package com.quorum.tessera.data.migration;
003
004import java.nio.file.Path;
005import java.nio.file.Paths;
006import java.util.Arrays;
007import java.util.Map;
008import java.util.Objects;
009import org.apache.commons.cli.CommandLine;
010import org.apache.commons.cli.CommandLineParser;
011import org.apache.commons.cli.DefaultParser;
012import org.apache.commons.cli.HelpFormatter;
013import org.apache.commons.cli.Option;
014import org.apache.commons.cli.Options;
015
016
017public class CmdLineExecutor {
018
019    private CmdLineExecutor() {
020        throw new UnsupportedOperationException("");
021    }
022
023
024        protected static int execute(String... args) throws Exception {
025
026        Options options = new Options();
027
028        options.addOption(
029                Option.builder()
030                        .longOpt("storetype")
031                        .desc("Store type i.e. bdb, dir, sqlite")
032                        .hasArg(true)
033                        .optionalArg(false)
034                        .numberOfArgs(1)
035                        .argName("TYPE").valueSeparator('=')
036                        .required()
037                        .build());
038
039        options.addOption(
040                Option.builder()
041                        .longOpt("inputpath")
042                        .desc("Path to input file or directory")
043                        .hasArg(true)
044                        .optionalArg(false)
045                        .numberOfArgs(1)
046                        .argName("PATH")
047                        .required()
048                        .build());
049
050        options.addOption(
051                Option.builder()
052                        .longOpt("exporttype")
053                        .desc("Export DB type i.e. h2, sqlite")
054                        .hasArg(true)
055                        .optionalArg(false)
056                        .numberOfArgs(1)
057                        .argName("TYPE")
058                        .required()
059                        .build());
060
061        options.addOption(
062                Option.builder()
063                        .longOpt("outputfile")
064                        .desc("Path to output file")
065                        .hasArg(true)
066                        .optionalArg(false)
067                        .numberOfArgs(1)
068                        .argName("PATH")
069                        .required()
070                        .build());
071
072        options.addOption(
073                Option.builder()
074                    .longOpt("dbuser")
075                    .desc("Database username to use")
076                    .hasArg(true)
077                    .optionalArg(true)
078                    .numberOfArgs(1)
079                    .argName("PATH")
080                    .required()
081                    .build());
082
083        options.addOption(
084                Option.builder()
085                    .longOpt("dbpass")
086                    .desc("Database password to use")
087                    .hasArg(true)
088                    .optionalArg(true)
089                    .numberOfArgs(1)
090                    .argName("PATH")
091                    .required()
092                    .build());
093
094        if (Arrays.asList(args).contains("help")) {
095            HelpFormatter formatter = new HelpFormatter();
096            formatter.printHelp("tessera-data-migration", options);
097            return 0;
098        }
099
100        final CommandLineParser parser = new DefaultParser();
101        final CommandLine line = parser.parse(options, args);
102
103        final StoreType storeType = StoreType.valueOf(line.getOptionValue("storetype").toUpperCase());
104        final StoreLoader storeLoader = StoreLoader.create(storeType);
105
106        final Path inputpath = Paths.get(line.getOptionValue("inputpath"));
107        final Map<byte[], byte[]> data = storeLoader.load(inputpath);
108
109        final String username = line.getOptionValue("dbuser");
110        final String password = line.getOptionValue("dbpass");
111
112        final String exportTypeStr = line.getOptionValue("exporttype");
113        final ExportType exportType = ExportType.valueOf(exportTypeStr.toUpperCase());
114
115        final Path outputFile = Paths.get(line.getOptionValue("outputfile")).toAbsolutePath();
116        final DataExporter dataExporter = DataExporterFactory.create(exportType);
117        dataExporter.export(data, outputFile, username, password);
118
119        System.out.printf("Exported data to %s", Objects.toString(outputFile));
120        System.out.println();
121
122        return 0;
123    }
124}
