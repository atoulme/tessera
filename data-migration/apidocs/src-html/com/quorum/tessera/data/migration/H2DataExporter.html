001package com.quorum.tessera.data.migration;
002
003import java.nio.file.Path;
004import java.sql.Connection;
005import java.sql.DriverManager;
006import java.sql.PreparedStatement;
007import java.sql.SQLException;
008import java.sql.Statement;
009import java.util.Map;
010import java.util.Map.Entry;
011
012public class H2DataExporter implements DataExporter {
013
014    private static final String INSERT_ROW = "INSERT INTO ENCRYPTED_TRANSACTION (HASH,ENCODED_PAYLOAD) VALUES (?,?)";
015
016    private static final String CREATE_TABLE = "CREATE TABLE ENCRYPTED_TRANSACTION "
017        + "(ENCODED_PAYLOAD LONGVARBINARY NOT NULL, "
018        + "HASH LONGVARBINARY NOT NULL UNIQUE, PRIMARY KEY (HASH))";
019
020    @Override
021    public void export(final Map<byte[], byte[]> data,
022                       final Path output,
023                       final String username,
024                       final String password) throws SQLException {
025
026        final String connectionString = "jdbc:h2:" + output.toString();
027
028        try (Connection conn = DriverManager.getConnection(connectionString, username, password)) {
029
030            try (Statement stmt = conn.createStatement()) {
031                stmt.executeUpdate(CREATE_TABLE);
032            }
033
034            try (PreparedStatement insertStatement = conn.prepareStatement(INSERT_ROW)) {
035                for (Entry<byte[], byte[]> values : data.entrySet()) {
036                    insertStatement.setBytes(1, values.getKey());
037                    insertStatement.setBytes(2, values.getValue());
038                    insertStatement.execute();
039                }
040            }
041
042        }
043    }
044
045}
