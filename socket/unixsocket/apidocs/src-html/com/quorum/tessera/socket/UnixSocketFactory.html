001package com.quorum.tessera.socket;
002
003import java.io.IOException;
004import java.net.ServerSocket;
005import java.net.Socket;
006import java.nio.file.Path;
007import java.util.ServiceLoader;
008
009/**
010 * Interface for different implementations to create server sockets and bind
011 * to existing sockets as a client
012 */
013public interface UnixSocketFactory {
014
015    /**
016     * Creates a new server unix socket
017     *
018     * @param socketFile the location to create the socket file
019     * @return the server that will accept requests
020     * @throws IOException if the file already exists, or another IO failure occurs
021     */
022     ServerSocket createServerSocket(Path socketFile) throws IOException;
023
024    /**
025     * Binds to a server socket as a client, opening a connection the the server
026     *
027     * @param socketFile the location of the server socket file to connect to
028     * @return the socket connection to read/write to/from
029     * @throws IOException if there was a problem connecting to the server
030     */
031     Socket createSocket(Path socketFile) throws IOException;
032
033    /**
034     * Find an implementation of {@link UnixSocketFactory} from the service loader
035     * @return an implementation of this interface
036     */
037     static UnixSocketFactory create() {
038         return ServiceLoader.load(UnixSocketFactory.class).iterator().next();
039     }
040     
041}
