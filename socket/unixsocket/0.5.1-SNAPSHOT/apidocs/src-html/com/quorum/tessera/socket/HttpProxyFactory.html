001package com.quorum.tessera.socket;
002
003import com.quorum.tessera.config.ServerConfig;
004import com.quorum.tessera.config.SslConfig;
005import com.quorum.tessera.ssl.context.model.SSLContextProperties;
006import com.quorum.tessera.ssl.strategy.TrustMode;
007
008import javax.net.SocketFactory;
009import javax.net.ssl.SSLContext;
010import java.net.URI;
011
012/**
013 * Creates proxies to the local HTTP server, setting up connection
014 * configuration such as the target address and SSL settings.
015 */
016public class HttpProxyFactory {
017
018    private final URI serverUri;
019
020    private final SocketFactory socketFactory;
021
022    public HttpProxyFactory(final ServerConfig serverConfig) throws Exception {
023        this.serverUri = serverConfig.getServerUri();
024
025        if (serverConfig.isSsl()) {
026
027            final SslConfig sslConfig = serverConfig.getSslConfig();
028
029            final SSLContext sslContext = TrustMode.NONE.createSSLContext(
030                new SSLContextProperties(
031                    serverConfig.getServerUri().toString(),
032                    sslConfig.getClientKeyStore(),
033                    sslConfig.getClientKeyStorePassword(),
034                    sslConfig.getClientTlsKeyPath(),
035                    sslConfig.getClientTlsCertificatePath(),
036                    sslConfig.getClientTrustStore(),
037                    sslConfig.getClientTrustStorePassword(),
038                    sslConfig.getClientTrustCertificates(),
039                    sslConfig.getKnownServersFile()
040                )
041            );
042            
043            this.socketFactory = sslContext.getSocketFactory();
044        } else {
045            this.socketFactory = SocketFactory.getDefault();
046        }
047
048    }
049
050    /**
051     * Creates a new proxy to the server, whose connection hasn't yet been opened
052     *
053     * @return
054     */
055    public HttpProxy create() {
056        return new HttpProxy(serverUri, socketFactory);
057    }
058
059}
