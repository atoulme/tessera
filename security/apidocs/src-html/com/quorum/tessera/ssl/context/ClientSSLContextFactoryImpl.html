001package com.quorum.tessera.ssl.context;
002
003import com.quorum.tessera.config.SslConfig;
004import com.quorum.tessera.ssl.context.model.SSLContextProperties;
005import com.quorum.tessera.ssl.exception.TesseraSecurityException;
006import com.quorum.tessera.ssl.strategy.TrustMode;
007import org.bouncycastle.operator.OperatorCreationException;
008
009import javax.net.ssl.SSLContext;
010import java.io.IOException;
011import java.security.GeneralSecurityException;
012
013public class ClientSSLContextFactoryImpl implements ClientSSLContextFactory {
014
015    @Override
016    public SSLContext from(String address, SslConfig sslConfig) {
017
018        TrustMode trustMode = TrustMode
019            .getValueIfPresent(sslConfig.getClientTrustMode().name())
020            .orElse(TrustMode.NONE);
021
022        final SSLContextProperties properties = new SSLContextProperties(
023            address,
024            sslConfig.getClientKeyStore(),
025            sslConfig.getClientKeyStorePassword(),
026            sslConfig.getClientTlsKeyPath(),
027            sslConfig.getClientTlsCertificatePath(),
028            sslConfig.getClientTrustStore(),
029            sslConfig.getClientTrustStorePassword(),
030            sslConfig.getClientTrustCertificates(),
031            sslConfig.getKnownServersFile()
032        );
033
034        try {
035            return trustMode.createSSLContext(properties);
036        } catch (IOException | OperatorCreationException | GeneralSecurityException ex) {
037            throw new TesseraSecurityException(ex);
038        }
039    }
040}
