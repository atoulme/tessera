001package com.quorum.tessera.ssl.util;
002
003import org.bouncycastle.asn1.x500.X500Name;
004import org.bouncycastle.asn1.x509.Extension;
005import org.bouncycastle.asn1.x509.GeneralName;
006import org.bouncycastle.asn1.x509.GeneralNames;
007import org.bouncycastle.cert.X509CertificateHolder;
008import org.bouncycastle.cert.X509v3CertificateBuilder;
009import org.bouncycastle.cert.jcajce.JcaX509CertificateConverter;
010import org.bouncycastle.cert.jcajce.JcaX509v3CertificateBuilder;
011import org.bouncycastle.jce.provider.BouncyCastleProvider;
012import org.bouncycastle.operator.ContentSigner;
013import org.bouncycastle.operator.OperatorCreationException;
014import org.bouncycastle.operator.jcajce.JcaContentSignerBuilder;
015
016import java.io.IOException;
017import java.io.OutputStream;
018import java.math.BigInteger;
019import java.nio.file.Files;
020import java.nio.file.Path;
021import java.security.*;
022import java.security.cert.CertificateException;
023import java.security.cert.X509Certificate;
024import java.util.Calendar;
025import java.util.Date;
026
027public interface TlsUtils {
028
029
030    String ENCRYPTION = "RSA";
031    String COMMON_NAME_STRING = "CN=";
032    String SIGNATURE_ALGORITHM = "SHA512WithRSAEncryption";
033    String KEYSTORE_TYPE = "JKS";
034
035    // These localhost values are added in SAN names for dev. They can be removed for production
036    // if necessary
037    String LOCALHOST = "localhost";
038    String LOCALHOST_IP = "127.0.0.1";
039    String LOCALHOST_IP_2 = "0.0.0.0";
040
041    Provider provider = new BouncyCastleProvider();
042
043    default void generateKeyStoreWithSelfSignedCertificate(String address, Path privateKeyFile, String password)
044        throws NoSuchAlgorithmException, IOException, OperatorCreationException,
045        CertificateException, InvalidKeyException, NoSuchProviderException, SignatureException, KeyStoreException {
046
047        final SecureRandom secureRandom = new SecureRandom();
048
049        KeyPairGenerator keyGen = KeyPairGenerator.getInstance(ENCRYPTION);
050        keyGen.initialize(2048, secureRandom);
051        KeyPair keypair = keyGen.generateKeyPair();
052        final PublicKey publicKey = keypair.getPublic();
053        final PrivateKey privateKey = keypair.getPrivate();
054        final String cnString = address.replaceFirst("^(http[s]?://www\\.|http[s]?://|www\\.)","");
055        final X500Name commonName = new X500Name(COMMON_NAME_STRING + cnString);
056        Date startDate = new Date(System.currentTimeMillis());
057        Calendar calendar = Calendar.getInstance();
058        calendar.setTime(startDate);
059        calendar.add(Calendar.YEAR, 1);
060        Date endDate = calendar.getTime();
061
062        X509v3CertificateBuilder builder = new JcaX509v3CertificateBuilder(
063            commonName,
064            new BigInteger(64, secureRandom),
065            startDate,
066            endDate,
067            commonName,
068            publicKey);
069
070        GeneralName[] subjectAlternativeNames = new GeneralName[] {
071            new GeneralName(GeneralName.dNSName, LOCALHOST),
072            new GeneralName(GeneralName.dNSName, HostnameUtil.create().getHostName()),
073            new GeneralName(GeneralName.iPAddress, LOCALHOST_IP),
074            new GeneralName(GeneralName.iPAddress, LOCALHOST_IP_2),
075            new GeneralName(GeneralName.iPAddress, HostnameUtil.create().getHostIpAddress())
076        };
077
078        builder.addExtension(
079            Extension.subjectAlternativeName,
080            false,
081            new GeneralNames(subjectAlternativeNames));
082
083        ContentSigner contentSigner = new JcaContentSignerBuilder(SIGNATURE_ALGORITHM).build(privateKey);
084        X509CertificateHolder certHolder = builder.build(contentSigner) ;
085        X509Certificate certificate =
086            new JcaX509CertificateConverter().setProvider(provider).getCertificate(certHolder);
087
088        certificate.verify(publicKey);
089
090        KeyStore keyStore = KeyStore.getInstance(KEYSTORE_TYPE);
091        keyStore.load(null, null);
092        keyStore.setKeyEntry("tessera",privateKey, password.toCharArray(), new X509Certificate[]{certificate});
093
094        try(final OutputStream keyStoreFile = Files.newOutputStream(privateKeyFile)) {
095            keyStore.store(keyStoreFile, password.toCharArray());
096        }
097    }
098
099    static TlsUtils create(){
100        Security.addProvider(new BouncyCastleProvider());
101        return new TlsUtils() {
102        };
103    }
104
105
106}
