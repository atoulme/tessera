001package com.quorum.tessera.ssl.util;
002
003import javax.xml.bind.DatatypeConverter;
004import java.security.MessageDigest;
005import java.security.cert.CertificateException;
006import java.security.cert.X509Certificate;
007
008public interface CertificateUtil {
009
010    default String thumbPrint(final X509Certificate certificate) throws CertificateException {
011        try {
012            final byte[] encoded = certificate.getEncoded();
013            return DatatypeConverter.printHexBinary(
014                MessageDigest.getInstance("SHA-1").digest(encoded)).toLowerCase();
015        } catch (Exception ex) {
016            throw new CertificateException("Cannot generate thumbprint for this certificate. Cause by ", ex);
017        }
018    }
019
020    static CertificateUtil create() {
021        return new CertificateUtil() {
022        };
023    }
024}
