012
013import java.io.FileNotFoundException;
014import java.io.IOException;
015import java.io.InputStream;
016import java.io.OutputStream;
017import java.lang.management.ManagementFactory;
018import java.nio.charset.StandardCharsets;

026import static java.nio.file.StandardOpenOption.*;
027import static java.util.Collections.singletonList;

029import javax.validation.*;
030
031public class DefaultCliAdapter implements CliAdapter {
032
033    private static final Logger LOGGER = LoggerFactory.getLogger(DefaultCliAdapter.class);
034
035    private final KeyGeneratorFactory keyGeneratorFactory = KeyGeneratorFactory.newFactory();
036
037    private final Validator validator = Validation.byDefaultProvider()
038            .configure()
039            .ignoreXmlConfiguration()
040            .buildValidatorFactory()
041            .getValidator();
042
043    @Override
044    public CliResult execute(String... args) throws Exception {
045
046        Options options = new Options();
047        
048
049        options.addOption(
050                Option.builder("configfile")
051                        .desc("Path to node configuration file")
052                        .hasArg(true)
053                        .optionalArg(false)
054                        .numberOfArgs(1)
055                        .argName("PATH")
056                        .build());
057
058        //If keygen then we require the path to the private key config path
059        options.addOption(
060                Option.builder("keygen")
061                        .desc("Path to private key config for generation of missing key files")
062                        .hasArg(true)
063                        .optionalArg(true)
064                        .argName("PATH")
065                        .build());
066
067        options.addOption(
068                Option.builder("filename")
069                        .desc("Path to private key config for generation of missing key files")
070                        .hasArg(true)
071                        .optionalArg(true)
072                        .argName("PATH")
073                        .build());
074
075        options.addOption(
076                Option.builder("keygenconfig")
077                        .desc("Path to private key config for generation of missing key files")
078                        .hasArg(true)
079                        .optionalArg(true)
080                        .argName("PATH")
081                        .build());
082
083        options.addOption(
084                Option.builder("output")
085                        .desc("Generate updated config file with generated keys")
086                        .hasArg(true)
087                        .numberOfArgs(1)
088                        .build());
089
090        options.addOption(
091                Option.builder("pidfile")
092                        .desc("Path to pid file")
093                        .hasArg(true)
094                        .optionalArg(false)
095                        .numberOfArgs(1)
096                        .argName("PATH")
097                        .build());
098
099        Map<String, Class> overrideOptions = OverrideUtil.buildConfigOptions();
100
101        overrideOptions.entrySet().forEach(entry -> {
102
103            final String optionName = entry.getKey();
104
105            final boolean isCollection = entry.getValue().isArray();
106
107            Class optionType = entry.getValue();
108
109            Option.Builder optionBuilder = Option.builder()
110                    .longOpt(optionName)
111                    .desc(String.format("Override option for %s , type: %s", optionName, optionType.getSimpleName()));
112
113            if (isCollection) {
114                optionBuilder.hasArgs()
115                        .argName(optionType.getSimpleName().toUpperCase() + "...");
116            } else {
117                optionBuilder.hasArg()
118                        .argName(optionType.getSimpleName().toUpperCase());
119            }
120            options.addOption(optionBuilder.build());
121
122        });
123
124        final List<String> argsList = Arrays.asList(args);
125        if (argsList.contains("help") || argsList.isEmpty()) {
126            HelpFormatter formatter = new HelpFormatter();
127            formatter.setWidth(200);
128            formatter.printHelp("tessera -configfile <PATH> [-keygen <PATH>] [-pidfile <PATH>]", options);
129            return new CliResult(0, true, false, null);
130        }
131
132        final CommandLineParser parser = new DefaultParser();
133
134        try {
135
136            final CommandLine line = parser.parse(options, args);
137
138            final Config config = parseConfig(line);
139
140            overrideOptions.entrySet().forEach(dynEntry -> {
141                String optionName = dynEntry.getKey();
142                if (line.hasOption(optionName)) {
143                    String[] values = line.getOptionValues(optionName);
144                    LOGGER.debug("Setting : {} with value(s) {}", optionName, values);
145                    OverrideUtil.setValue(config, optionName, values);
146                    LOGGER.debug("Set : {} with value(s) {}", optionName, values);
147                }
148            });
149
150            if (Objects.nonNull(config)) {
151
152                Set<ConstraintViolation<Config>> violations = validator.validate(config);
153                if (!violations.isEmpty()) {
154                    throw new ConstraintViolationException(violations);
155                }
156            }
157
158            if (line.hasOption("pidfile")) {
159                createPidFile(line);
160            }
161
162            return new CliResult(0, false, line.hasOption("keygen"), config);
163
164        } catch (ParseException exp) {
165            throw new CliException(exp.getMessage());
166        }
167
168    }
169
170    private Config parseConfig(CommandLine commandLine) throws IOException {
171
172        final ConfigFactory configFactory = ConfigFactory.create();
173
174        final List<String> keyGenConfigs = this.getKeyGenConfig(commandLine);
175        final ArgonOptions options = this.keygenConfiguration(commandLine).orElse(null);
176
177        Config config = null;
178
179        if (commandLine.hasOption("configfile")) {
180            final Path path = Paths.get(commandLine.getOptionValue("configfile"));
181
182            if (!Files.exists(path)) {
183                throw new FileNotFoundException(String.format("%s not found.", path));
184            }
185
186            try (InputStream in = Files.newInputStream(path)) {
187                config = configFactory.create(in, options, keyGenConfigs.toArray(new String[0]));
188            }
189
190            if (!keyGenConfigs.isEmpty()) {
191                //we have generated new keys, so we need to output the new configuration
192                output(commandLine, config);
193            }
194
195        } else if(commandLine.hasOption("keygen")) {
196            final KeyGenerator generator = keyGeneratorFactory.create();
197            keyGenConfigs
198                    .stream()
199                    .map(name -> generator.generate(name, options))
200                    .collect(Collectors.toList());
201        } else {
202            throw new CliException("One or both: -configfile <PATH> or -keygen options are required.");
203        }
204
205        return config;
206    }
207
208    private List<String> getKeyGenConfig(CommandLine commandLine) {
209
210        if (commandLine.hasOption("keygen")) {
211
212            if (commandLine.hasOption("filename")) {
213
214                final String keyNames = commandLine.getOptionValue("filename");
215                if (keyNames != null) {
216                    return Stream.of(keyNames.split(",")).collect(Collectors.toList());
217                }
218
219            }
220
221            return singletonList("");
222        }
223
224        return new ArrayList<>();
225    }
226
227    private static Optional<ArgonOptions> keygenConfiguration(final CommandLine commandLine) throws IOException {
228        if (commandLine.hasOption("keygenconfig")) {
229            final String pathName = commandLine.getOptionValue("keygenconfig");
230            final InputStream configStream = Files.newInputStream(Paths.get(pathName));
231
232            ArgonOptions argonOptions = JaxbUtil.unmarshal(configStream, ArgonOptions.class);
233            return Optional.of(argonOptions);
234        }
235        return Optional.empty();
236    }
237
238    private static void output(CommandLine commandLine, Config config) throws IOException {
239
240        if (commandLine.hasOption("output")) {
241            final Path outputConfigFile = Paths.get(commandLine.getOptionValue("output"));
242
243            try (OutputStream out = Files.newOutputStream(outputConfigFile, CREATE_NEW)) {
244                JaxbUtil.marshal(config, out);
245            }
246        } else {
247            JaxbUtil.marshal(config, System.out);
248        }
249    }
250
251    private static void createPidFile(CommandLine commandLine) throws IOException {
252
253        final Path pidFilePath = Paths.get(commandLine.getOptionValue("pidfile"));
254
255        if (Files.exists(pidFilePath)) {
256            LOGGER.info("File already exists {}", pidFilePath);
257        } else {
258            Files.createFile(pidFilePath);
259            LOGGER.info("Created pid file {}", pidFilePath);
260        }
261
262        final String pid = ManagementFactory.getRuntimeMXBean().getName().split("@")[0];
263
264        try (final OutputStream stream = Files.newOutputStream(pidFilePath, CREATE, TRUNCATE_EXISTING)) {
265            stream.write(pid.getBytes(StandardCharsets.UTF_8));
266        }
267    }
