030    Logger LOGGER = LoggerFactory.getLogger(OverrideUtil.class);
031
032    List<Class> SIMPLE_TYPES = Collections.unmodifiableList(
033            Arrays.asList(String.class,Path.class, Integer.class, Boolean.class, Long.class));
034
035    Map<Class<?>, Class<?>> PRIMATIVE_LOOKUP = Collections.unmodifiableMap(new HashMap<Class<?>, Class<?>>() {
036        {
037            put(Boolean.TYPE, Boolean.class);
038            put(Byte.TYPE, Byte.class);
039            put(Character.TYPE, Character.class);
040            put(Short.TYPE, Short.class);
041            put(Integer.TYPE, Integer.class);
042            put(Long.TYPE, Long.class);
043            put(Double.TYPE, Double.class);
044            put(Float.TYPE, Float.class);
045            put(Void.TYPE, Void.TYPE);
046        }
047    });

049    static Map<String, Class> buildConfigOptions() {
050        return fields(null, Config.class);
051    }

053    static String resolveName(Field field) {
054        if (!field.isAnnotationPresent(XmlElement.class)) {
055            return field.getName();
056        }
057        final XmlElement xmlElement = field.getAnnotation(XmlElement.class);
058        String name = xmlElement.name();
059        return Objects.equals("##default", name) ? field.getName() : name;
060    }
061
062    static Class resolveCollectionParameter(Type type) {
063
064        return Stream.of(type)
065                .map(ParameterizedType.class::cast)
066                .map(pt -> pt.getActualTypeArguments()[0])
067                .map(Type::getTypeName)
068                .map(n -> ReflectCallback.execute(() -> Class.forName(n))).findAny().get();
069
070    }
071
072    static boolean isSimple(Field field) {
073        return isSimple(field.getType());
074    }
075
076    static boolean isSimple(Class type) {
077
078        if (type.isPrimitive()) {
079            return true;
080        }
081
082        if (type.getPackage() == null) {
083            return true;
084        }
085
086        if (SIMPLE_TYPES.contains(type)) {
087            return true;
088        }
089
090        return type.isEnum();
091    }
092
093    static Map<String, Class> fields(String prefix, Class type) {
094
095        String amendedPrefix = Optional.ofNullable(prefix)
096                .map(s -> s.concat("."))
097                .orElse("");
098
099        Map<String, Class> list = new HashMap<>();
100        for (Field field : type.getDeclaredFields()) {
101
102            if (isSimple(field)) {
103
104                list.put(amendedPrefix + resolveName(field), field.getType());
105                continue;
106            }
107
108            if (field.getType().getPackage().equals(Config.class.getPackage())) {
109                list.putAll(fields(amendedPrefix + resolveName(field), field.getType()));
110            }
111
112            if (Collection.class.isAssignableFrom(field.getType())) {
113                Class t = resolveCollectionParameter(field.getGenericType());
114
115                if (isSimple(t)) {
116                    final Class arrayType = toArrayType(t);
117                    list.put(amendedPrefix + resolveName(field), arrayType);
118
119                } else {
120                    list.putAll(fields(amendedPrefix + resolveName(field), t));
121                }
122            }
123        }
124
125        return list;
126
127    }
128
129    static <T> Class<T[]> toArrayType(Class<T> t) {
130        return (Class<T[]>) Array.newInstance(t, 0).getClass();
131    }

133    /**
134     * Directly set field values using relection.
135     *
136     * @param root
137     * @param path
138     * @param value
139     */
140    static void setValue(Object root, String path, String... value) {
141
142        final ListIterator<String> pathTokens = Arrays.asList(path.split("\\.")).listIterator();
143
144        final Class rootType = root.getClass();
145
146        while (pathTokens.hasNext()) {
147
148            final String token = pathTokens.next();
149            final Field field = resolveField(rootType, token);
150            field.setAccessible(true);
151
152            final Class fieldType = field.getType();
153
154            if (Collection.class.isAssignableFrom(fieldType)) {
155
156                final Class genericType = resolveCollectionParameter(field.getGenericType());
157
158                List list = (List) Optional.ofNullable(getValue(root, field))
159                        .orElse(new ArrayList<>());
160                if (isSimple(genericType)) {
161
162                    List convertedValues = (List) Stream.of(value)
163                            .map(v -> convertTo(genericType, v))
164                            .collect(Collectors.toList());
165
166                    setValue(root, field, convertedValues);
167
168                } else {
169
170                    List<String> builder = new ArrayList<>();
171                    pathTokens.forEachRemaining(builder::add);
172                    String nestedPath = builder.stream().collect(Collectors.joining("."));
173
174                    final Object[] newList = Arrays.copyOf(list.toArray(), value.length);
175
176                    for (int i = 0; i < value.length; i++) {
177                        final String v = value[i];
178
179                        final Object nestedObject = Optional.ofNullable(newList[i])
180                                .orElse(createInstance(genericType));
181
182                        initialiseNestedObjects(nestedObject);
183
184                        setValue(nestedObject, nestedPath, v);
185                        newList[i] = nestedObject;
186                    }
187                    setValue(root, field, Arrays.asList(newList));
188
189                }
190
191            } else if (isSimple(fieldType)) {
192                Class convertedType = PRIMATIVE_LOOKUP.getOrDefault(fieldType, fieldType);
193                Object convertedValue = convertTo(convertedType, value[0]);
194                setValue(root, field, convertedValue);
195
196            } else {
197
198                Object nestedObject = getOrCreate(root, field);
199                List<String> builder = new ArrayList<>();
200                pathTokens.forEachRemaining(builder::add);
201                String nestedPath = builder.stream().collect(Collectors.joining("."));
202                setValue(nestedObject, nestedPath, value);
203
204                setValue(root, field, nestedObject);
205            }
206
207        }
208
209    }
210
211    static <T> T getOrCreate(Object from, Field field) {
212        T value = getValue(from, field);
213        return Optional.ofNullable(value)
214                .orElse((T) createInstance(field.getType()));
215    }
216
217    static <T> T getValue(Object from, Field field) {
218        return ReflectCallback.execute(() -> {
219            return (T) field.get(from);
220        });
221    }
222
223    static void setValue(Object obj, Field field, Object value) {
224        ReflectCallback.execute(() -> {
225            field.set(obj, value);
226            return null;
227        });
228    }
229
230    static Field resolveField(Class type, String name) {
231        LOGGER.debug("Resolving {}#{}", type, name);
232
233        return Stream.of(type.getDeclaredFields())
234                .filter(f -> f.isAnnotationPresent(XmlElement.class))
235                .filter(f -> f.getAnnotation(XmlElement.class).name().equals(name))
236                .findAny()
237                .orElseGet(() -> ReflectCallback.execute(() -> type.getDeclaredField(name)));
238    }
239
240    static <T> T createInstance(Class<T> type) {
241
242        return ReflectCallback.execute(() -> {
243            Method factoryMethod = type.getDeclaredMethod("create");
244            factoryMethod.setAccessible(true);
245            final Object instance = factoryMethod.invoke(null);
246            initialiseNestedObjects(instance);
247            return (T) instance;
248        });
249
250    }
251
252    static Class classForName(String classname) {
253        return ReflectCallback.execute(() -> Class.forName(classname));
254    }
255
256    static void initialiseNestedObjects(Object obj) {
257        ReflectCallback.execute(() -> {
258            Class type = obj.getClass();
259            Field[] fields = type.getDeclaredFields();
260
261            for (Field field : fields) {
262                field.setAccessible(true);
263                Class fieldType = field.getType();
264                if (isSimple(fieldType)) {
265                    continue;
266                }
267
268                if (Collection.class.isAssignableFrom(fieldType)) {
269                    setValue(obj, field, new ArrayList<>());
270                    continue;
271                }
272
273                Object nestedObject = createInstance(fieldType);
274                initialiseNestedObjects(nestedObject);
275                setValue(obj, field, nestedObject);
276
277            }
278            return null;
279        });
280
281    }
282
283    static <T> T convertTo(Class<T> type, String value) {
284
285        if (Objects.isNull(value)) {
286            return null;
287        }
288
289        if (String.class.equals(type)) {
290            return (T) value;
291        }
292
293        if (Path.class.equals(type)) {
294            return (T) Paths.get(value);
295        }
296
297        if (type.isEnum()) {
298            return (T) Enum.valueOf(type.asSubclass(Enum.class), value);
299        }
300
301        return SIMPLE_TYPES.stream()
302                .filter(t -> t.equals(type))
303                .findFirst()
304                .map(c -> ReflectCallback.execute(() -> c.getDeclaredMethod("valueOf", String.class)))
305                .map(m -> ReflectCallback.execute(() -> m.invoke(null, value)))
306                .map(type::cast)
307                .get();
308
309    }
