002package com.quorum.tessera.config.cli;
003
004import com.quorum.tessera.config.Config;
005import java.util.Objects;
006import java.util.Optional;
007
008
009public class CliResult {
010    
011    private final Integer status;
012    private final boolean isHelpOn;
013    private final boolean isKeyGenOn;
014    private final Config config;
015
016    public CliResult(Integer status, boolean isHelpOn, boolean isKeyGenOn, Config config) {
017        this.status = Objects.requireNonNull(status);
018        this.isHelpOn = isHelpOn;
019        this.isKeyGenOn = isKeyGenOn;
020        this.config = config;
021    }
022
023    public Integer getStatus() {
024        return status;
025    }
026
027    public boolean isHelpOn() {
028        return isHelpOn;
029    }
030
031    public boolean isKeyGenOn() {return isKeyGenOn;}
032
033    public Optional<Config> getConfig() {
034        return Optional.ofNullable(config);
035    }
036}
