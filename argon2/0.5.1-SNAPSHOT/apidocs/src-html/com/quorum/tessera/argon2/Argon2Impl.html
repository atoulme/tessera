001package com.quorum.tessera.argon2;
002
003import de.mkammerer.argon2.Argon2Advanced;
004import de.mkammerer.argon2.Argon2Factory;
005import org.slf4j.Logger;
006import org.slf4j.LoggerFactory;
007
008import java.util.Arrays;
009import java.util.stream.Stream;
010
011public class Argon2Impl implements Argon2 {
012
013    private static final Logger LOGGER = LoggerFactory.getLogger(Argon2Impl.class);
014
015    private static final ArgonOptions DEFAULT_OPTIONS = new ArgonOptions("i", 10, 1048576, 4);
016
017    @Override
018    public ArgonResult hash(final ArgonOptions options, final String password, final byte[] salt) {
019        final Argon2Advanced argon2 = this.getArgon2Instance(options.getAlgorithm());
020
021        final byte[] hash = argon2.rawHash(
022            options.getIterations(),
023            options.getMemory(),
024            options.getParallelism(),
025            password,
026            salt
027        );
028
029        LOGGER.debug("Argon2 hash produced the array {}", Arrays.toString(hash));
030
031        final String algoName = Stream.of("d", "id", "i")
032            .filter(options.getAlgorithm()::equals)
033            .findAny()
034            .orElse("i");
035
036        return new ArgonResult(
037            new ArgonOptions(
038                algoName,
039                options.getIterations(),
040                options.getMemory(),
041                options.getParallelism()
042            ),
043            salt,
044            hash
045        );
046    }
047
048    @Override
049    public ArgonResult hash(final String password, final byte[] salt) {
050        return this.hash(DEFAULT_OPTIONS, password, salt);
051    }
052
053    /**
054     * The string form of the algorithm to use.
055     * If an invalid algorithm is chosen, a default of Argon2i is chosen.
056     *
057     * @param algorithm the algorithm to use
058     * @return an instance of the chosen algorithm
059     */
060    private Argon2Advanced getArgon2Instance(final String algorithm) {
061        LOGGER.debug("Searching for the Argon2 algorithm {}", algorithm);
062
063        switch (algorithm) {
064            case "d":
065                return Argon2Factory.createAdvanced(Argon2Factory.Argon2Types.ARGON2d);
066            case "id":
067                return Argon2Factory.createAdvanced(Argon2Factory.Argon2Types.ARGON2id);
068            case "i":
069            default:
070                return Argon2Factory.createAdvanced(Argon2Factory.Argon2Types.ARGON2i);
071        }
072    }
073}
