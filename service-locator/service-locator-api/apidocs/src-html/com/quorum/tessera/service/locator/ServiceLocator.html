001package com.quorum.tessera.service.locator;
002
003import java.util.ServiceLoader;
004import java.util.Set;
005
006/**
007 * Creates a set of services that are configured
008 */
009public interface ServiceLocator {
010
011    /**
012     * Retrieves all the services specified in the configuration file
013     *
014     * @param filename the file from which to load service definitions from
015     * @return the set of all configuration services
016     */
017    Set<Object> getServices(String filename);
018
019    /**
020     * Returns an implementation of the {@link ServiceLocator} from the service loader
021     *
022     * @return the {@link ServiceLocator} instance
023     */
024    static ServiceLocator create() {
025        return ServiceLoader.load(ServiceLocator.class).iterator().next();
026    }
027
028}
