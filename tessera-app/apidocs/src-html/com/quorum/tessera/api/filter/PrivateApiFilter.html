001package com.quorum.tessera.api.filter;
002
003import com.quorum.tessera.config.ServerConfig;
004import org.slf4j.Logger;
005import org.slf4j.LoggerFactory;
006
007import javax.servlet.http.HttpServletRequest;
008import javax.ws.rs.container.ContainerRequestContext;
009import javax.ws.rs.container.ContainerRequestFilter;
010import javax.ws.rs.core.Context;
011import javax.ws.rs.core.Response;
012
013@PrivateApi
014public class PrivateApiFilter implements ContainerRequestFilter {
015
016    private static final Logger LOGGER = LoggerFactory.getLogger(PrivateApiFilter.class);
017
018    private HttpServletRequest httpServletRequest;
019
020    private final String hostname;
021
022    public PrivateApiFilter(final ServerConfig serverConfig) {
023        this.hostname = serverConfig.getServerUri().getHost();
024    }
025
026    /**
027     * Extract the callers hostname and address,
028     * and check it against our configured hostname to check they are the same
029     * <p>
030     * If the host is not the same as ours, finish the filter chain here and return an Unauthorized response
031     *
032     * @param requestContext the context of the current request
033     */
034    @Override
035    public void filter(final ContainerRequestContext requestContext) {
036
037        if(this.httpServletRequest == null) {
038            LOGGER.debug("No servlet available, could not determine request origin. Allowing...");
039            return;
040        }
041
042        final String remoteAddress = httpServletRequest.getRemoteAddr();
043        final String remoteHost = httpServletRequest.getRemoteHost();
044
045        final boolean allowed = hostname.equals(remoteAddress) || hostname.equals(remoteHost);
046
047        if (!allowed) {
048            requestContext.abortWith(Response.status(Response.Status.UNAUTHORIZED).build());
049        }
050
051    }
052
053    /**
054     * Apply the current HTTP context to the filter, to check the remote host
055     *
056     * @param request the request to be filtered
057     */
058    @Context
059    public void setHttpServletRequest(final HttpServletRequest request) {
060        this.httpServletRequest = request;
061    }
062}
