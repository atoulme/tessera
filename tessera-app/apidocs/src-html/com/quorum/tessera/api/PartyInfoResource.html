001package com.quorum.tessera.api;
002
003import com.quorum.tessera.node.PartyInfoParser;
004import com.quorum.tessera.node.PartyInfoService;
005import com.quorum.tessera.node.model.PartyInfo;
006import io.swagger.annotations.ApiOperation;
007import io.swagger.annotations.ApiParam;
008import io.swagger.annotations.ApiResponse;
009import io.swagger.annotations.ApiResponses;
010
011import javax.ws.rs.Consumes;
012import javax.ws.rs.POST;
013import javax.ws.rs.Path;
014import javax.ws.rs.Produces;
015import javax.ws.rs.core.MediaType;
016import javax.ws.rs.core.Response;
017import javax.ws.rs.core.StreamingOutput;
018
019import static java.util.Objects.requireNonNull;
020
021/**
022 * Defines endpoints for requesting node discovery (partyinfo) information
023 */
024@Path("/partyinfo")
025public class PartyInfoResource {
026
027    private final PartyInfoParser partyInfoParser;
028
029    private final PartyInfoService partyInfoService;
030
031    public PartyInfoResource(final PartyInfoService partyInfoService,
032                             final PartyInfoParser partyInfoParser) {
033        this.partyInfoService = requireNonNull(partyInfoService, "partyInfoService must not be null");
034        this.partyInfoParser = requireNonNull(partyInfoParser, "partyInfoParser must not be null");
035    }
036
037    /**
038     * Allows node information to be retrieved in a specific encoded form
039     * including other node URLS and public key to URL mappings
040     *
041     * @param payload The encoded node information from the requester
042     * @return the merged node information from this node, which may contain new information
043     */
044    @POST
045    @Consumes(MediaType.APPLICATION_OCTET_STREAM)
046    @Produces(MediaType.APPLICATION_OCTET_STREAM)
047    @ApiOperation(value = "Request public key/url of other nodes", produces = "public keylist/url")
048    @ApiResponses({@ApiResponse(code = 200, message = "Encoded PartyInfo Data", response = byte[].class)})
049    public Response partyInfo(@ApiParam(required = true) final byte[] payload) {
050
051        final PartyInfo partyInfo = partyInfoParser.from(payload);
052
053        final PartyInfo updatedPartyInfo = partyInfoService.updatePartyInfo(partyInfo);
054
055        final byte[] encoded = partyInfoParser.to(updatedPartyInfo);
056
057        final StreamingOutput streamingOutput = out -> out.write(encoded);
058
059        return Response.status(Response.Status.OK)
060            .entity(streamingOutput)
061            .build();
062    }
063}
