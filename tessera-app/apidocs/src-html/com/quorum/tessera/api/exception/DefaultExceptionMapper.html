001package com.quorum.tessera.api.exception;
002
003import javax.ws.rs.core.Response;
004import javax.ws.rs.core.Response.Status;
005import javax.ws.rs.ext.ExceptionMapper;
006import javax.ws.rs.ext.Provider;
007import org.slf4j.Logger;
008import org.slf4j.LoggerFactory;
009
010@Provider
011public class DefaultExceptionMapper implements ExceptionMapper<Exception> {
012    
013    private static final Logger LOGGER = LoggerFactory.getLogger(DefaultExceptionMapper.class);
014    
015    @Override
016    public Response toResponse(Exception exception) {
017        
018        LOGGER.error("{}",exception.getMessage());
019        
020        return Response.status(Status.INTERNAL_SERVER_ERROR)
021                .entity(exception.getMessage())
022                .build();
023    }
024}
