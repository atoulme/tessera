001package com.quorum.tessera.node;
002
003import com.quorum.tessera.api.model.ApiPath;
004import com.quorum.tessera.node.model.Party;
005import com.quorum.tessera.node.model.PartyInfo;
006import com.quorum.tessera.sync.ResendPartyStore;
007import org.slf4j.Logger;
008import org.slf4j.LoggerFactory;
009
010import java.net.ConnectException;
011import java.util.Objects;
012import java.util.Set;
013
014/**
015 * Polls every so often to all known nodes for any new discoverable nodes
016 * This keeps all nodes up-to date and discoverable by other nodes
017 */
018public class PartyInfoPoller implements Runnable {
019
020    private static final Logger LOGGER = LoggerFactory.getLogger(PartyInfoPoller.class);
021
022    private final PartyInfoService partyInfoService;
023
024    private final PostDelegate postDelegate;
025
026    private final PartyInfoParser partyInfoParser;
027
028    private final ResendPartyStore resendPartyStore;
029
030    public PartyInfoPoller(final PartyInfoService partyInfoService,
031                           final PartyInfoParser partyInfoParser,
032                           final PostDelegate postDelegate,
033                           final ResendPartyStore resendPartyStore) {
034        this.partyInfoService = Objects.requireNonNull(partyInfoService);
035        this.partyInfoParser = Objects.requireNonNull(partyInfoParser);
036        this.postDelegate = Objects.requireNonNull(postDelegate);
037        this.resendPartyStore = Objects.requireNonNull(resendPartyStore);
038    }
039
040    /**
041     * Iterates over all known parties and contacts them for the current state
042     * of their known node discovery list
043     *
044     * It then updates this nodes list of data with any new information collected
045     */
046    @Override
047    public void run() {
048        LOGGER.debug("Polling {}", getClass().getSimpleName());
049
050        final PartyInfo partyInfo = partyInfoService.getPartyInfo();
051
052        final byte[] encodedPartyInfo = partyInfoParser.to(partyInfo);
053
054        partyInfo
055            .getParties()
056            .parallelStream()
057            .filter(party -> !party.getUrl().equals(partyInfo.getUrl()))
058            .map(Party::getUrl)
059            .map(url -> pollSingleParty(url, encodedPartyInfo))
060            .filter(Objects::nonNull)
061            .map(partyInfoParser::from)
062            .forEach(newPartyInfo -> {
063                this.resendForNewKeys(newPartyInfo);
064                partyInfoService.updatePartyInfo(newPartyInfo);
065            });
066
067        LOGGER.debug("Polled {}. PartyInfo : {}", getClass().getSimpleName(), partyInfo);
068    }
069
070    /**
071     * Sends a request for node information to a single target
072     * If it cannot connect to the target, it returns null, otherwise throws
073     * any exception that can be thrown from {@link javax.ws.rs.client.Client}
074     *
075     * @param url              the target URL to call
076     * @param encodedPartyInfo the encoded current party information
077     * @return the encoded partyinfo from the target node, or null is the node
078     * could not be reached
079     */
080    private byte[] pollSingleParty(final String url, final byte[] encodedPartyInfo) {
081
082        try {
083            return postDelegate.doPost(url, ApiPath.PARTYINFO, encodedPartyInfo);
084
085        } catch (final Exception ex) {
086
087            if (ConnectException.class.isInstance(ex.getCause())) {
088                LOGGER.warn("Server error {} when connecting to {}", ex.getMessage(), url);
089                LOGGER.debug(null, ex);
090                return null;
091            } else {
092                LOGGER.error("Error thrown while executing poller. ", ex);
093                throw ex;
094            }
095
096        }
097
098    }
099
100    private void resendForNewKeys(final PartyInfo receivedPartyInfo) {
101        final Set<Party> newPartiesFound = this.partyInfoService.findUnsavedParties(receivedPartyInfo);
102        this.resendPartyStore.addUnseenParties(newPartiesFound);
103    }
104
105}
