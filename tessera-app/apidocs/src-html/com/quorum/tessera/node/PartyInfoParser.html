001package com.quorum.tessera.node;
002
003import com.quorum.tessera.nacl.Key;
004import com.quorum.tessera.node.model.Party;
005import com.quorum.tessera.node.model.PartyInfo;
006import com.quorum.tessera.node.model.Recipient;
007import com.quorum.tessera.util.BinaryEncoder;
008
009import java.nio.ByteBuffer;
010import java.util.Arrays;
011import java.util.HashSet;
012import java.util.List;
013import java.util.Set;
014import java.util.stream.Collectors;
015
016import static java.nio.charset.StandardCharsets.UTF_8;
017
018/**
019 * A parser for PartyInfo node discovery information
020 */
021public interface PartyInfoParser extends BinaryEncoder {
022
023    /**
024     * Decodes a set of PartyInfo to the format that is shared between nodes
025     *
026     * @param encoded the encoded information that needs to be read
027     * @return the decoded {@link PartyInfo} which contains the other nodes
028     * information
029     */
030    default PartyInfo from(final byte[] encoded) {
031
032        final ByteBuffer byteBuffer = ByteBuffer.wrap(encoded);
033
034        final int urlLength = (int) byteBuffer.getLong();
035
036        final byte[] urlBytes = new byte[urlLength];
037        byteBuffer.get(urlBytes);
038        final String url = new String(urlBytes, UTF_8);
039
040        final int numberOfRecipients = (int) byteBuffer.getLong();
041
042        final Set<Recipient> recipients = new HashSet<>();
043
044        for (int i = 0; i < numberOfRecipients; i++) {
045
046            final int recipientElementCount = (int) byteBuffer.getLong();
047
048            final int recipientKeyLength = (int) byteBuffer.getLong();
049            final byte[] recipientKeyBytes = new byte[recipientKeyLength];
050            byteBuffer.get(recipientKeyBytes);
051
052            final int recipientUrlValueLength = (int) byteBuffer.getLong();
053            final byte[] urlValueData = new byte[recipientUrlValueLength];
054            byteBuffer.get(urlValueData);
055            final String recipientUrl = new String(urlValueData, UTF_8);
056
057            recipients.add(new Recipient(new Key(recipientKeyBytes), recipientUrl));
058
059        }
060
061        final int partyCount = (int) byteBuffer.getLong();
062
063        final Party[] parties = new Party[partyCount];
064        for (int i = 0; i < partyCount; i++) {
065            long partyElementLength = byteBuffer.getLong();
066            byte[] ptyData = new byte[(int) partyElementLength];
067            byteBuffer.get(ptyData);
068            String ptyURL = new String(ptyData, UTF_8);
069            parties[i] = new Party(ptyURL);
070        }
071
072        return new PartyInfo(url, recipients, new HashSet<>(Arrays.asList(parties)));
073    }
074
075    /**
076     * Encodes a {@link PartyInfo} object to the defined structure that is
077     * shared between nodes
078     *
079     * The result can be feed into {@link PartyInfoParser#from(byte[])} to
080     * produce the input to this function.
081     *
082     * @param partyInfo the information to encode
083     * @return the encoded result that should be shared with other nodes
084     */
085    default byte[] to(final PartyInfo partyInfo) {
086
087        //prefix and url bytes
088        final byte[] url = encodeField(partyInfo.getUrl().getBytes(UTF_8));
089
090        //each element in the list is one encoded element from the map
091        //so the prefix is always 2 (2 elements) and
092        final List<byte[]> recipients = partyInfo.getRecipients()
093            .stream()
094            .map(r -> new byte[][]{
095                    r.getKey().getKeyBytes(),
096                    r.getUrl().getBytes(UTF_8)
097                }
098            ).map(this::encodeArray)
099            .collect(Collectors.toList());
100        final int recipientLength = recipients.stream().mapToInt(r -> r.length).sum();
101
102        final List<byte[]> parties = partyInfo.getParties()
103            .stream()
104            .map(p -> p.getUrl().getBytes(UTF_8))
105            .collect(Collectors.toList());
106
107        final byte[] partiesBytes = encodeArray(parties);
108
109        final ByteBuffer byteBuffer = ByteBuffer
110            .allocate(url.length + Long.BYTES + recipientLength + partiesBytes.length)
111            .put(url)
112            .putLong(partyInfo.getRecipients().size());
113
114        recipients.forEach(byteBuffer::put);
115        byteBuffer.put(partiesBytes);
116
117        return byteBuffer.array();
118
119    }
120
121    /**
122     * Creates a new parser with default settings
123     *
124     * @return a default parser
125     */
126    static PartyInfoParser create() {
127        return new PartyInfoParser() {
128        };
129    }
130
131}
