001package com.quorum.tessera.node;
002
003import com.quorum.tessera.config.ServerConfig;
004import com.quorum.tessera.ssl.context.SSLContextFactory;
005
006import javax.net.ssl.SSLContext;
007import javax.ws.rs.client.Client;
008import javax.ws.rs.client.ClientBuilder;
009import java.util.Objects;
010
011/**
012 * Creates HTTP clients that can make requests
013 * <p>
014 * Makes the client SSL enabled if specified
015 */
016public class ClientFactory {
017
018    private final SSLContextFactory sslContextFactory;
019
020    public ClientFactory(final SSLContextFactory sslContextFactory) {
021        this.sslContextFactory = Objects.requireNonNull(sslContextFactory);
022    }
023
024    /**
025     * Creates a new client, which may or may not be SSL enabled depending on
026     * the configuration
027     *
028     * @see Client
029     */
030    public Client buildFrom(final ServerConfig config) {
031        if (config.isSsl()) {
032            final SSLContext sslContext = sslContextFactory.from(
033                config.getServerUri().toString(),
034                config.getSslConfig());
035            return ClientBuilder.newBuilder()
036                .sslContext(sslContext)
037                .build();
038        } else {
039            return ClientBuilder.newClient();
040        }
041    }
042
043}
