001package com.quorum.tessera.node.model;
002
003import java.util.Collections;
004import java.util.HashSet;
005import java.util.Objects;
006import java.util.Set;
007
008/**
009 * Contains all information that is transferred between two nodes on the
010 * network, including:
011 * - the external URL of this node
012 * - all known {@link Recipient} that contains public key to URL mappings
013 * - all known URLs on the network
014 */
015public class PartyInfo {
016    
017    private final String url;
018
019    private Set<Recipient> recipients;
020
021    private Set<Party> parties;
022
023    public PartyInfo(final String url, final Set<Recipient> recipients, final Set<Party> parties) {
024        this.url = Objects.requireNonNull(url);
025        this.recipients = Collections.unmodifiableSet(new HashSet<>(recipients));
026        this.parties = Collections.unmodifiableSet(new HashSet<>(parties));
027    }
028
029    public String getUrl() {
030        return url;
031    }
032
033    public Set<Recipient> getRecipients() {
034        return recipients;
035    }
036
037    public Set<Party> getParties() {
038        return parties;
039    }
040
041}
