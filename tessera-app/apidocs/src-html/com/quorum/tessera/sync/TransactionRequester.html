001package com.quorum.tessera.sync;
002
003/**
004 * Makes requests to other nodes to resend their transactions
005 * <p>
006 * Handles creating the correct entity and retrying on a failed attempt
007 */
008public interface TransactionRequester {
009
010    int MAX_ATTEMPTS = 5;
011
012    /**
013     * Makes a request to the given node to resend transactions for
014     *
015     * @param url the URL to contact for resending
016     * @return whether all the resend requests for all keys was successful or not
017     */
018    boolean requestAllTransactionsFromNode(String url);
019
020}
