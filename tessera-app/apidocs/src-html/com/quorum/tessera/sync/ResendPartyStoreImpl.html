001package com.quorum.tessera.sync;
002
003import com.quorum.tessera.config.Config;
004import com.quorum.tessera.config.Peer;
005import com.quorum.tessera.node.model.Party;
006import com.quorum.tessera.sync.model.SyncableParty;
007
008import java.util.*;
009
010import static java.util.stream.Collectors.toSet;
011
012/**
013 * An in-memory store of outstanding parties to contact for transaction resending
014 */
015public class ResendPartyStoreImpl implements ResendPartyStore {
016
017    private Queue<SyncableParty> outstandingParties;
018
019    public ResendPartyStoreImpl(final Config config) {
020        this.outstandingParties = new LinkedList<>();
021
022        final Set<Party> initialParties = config
023            .getPeers()
024            .stream()
025            .map(Peer::getUrl)
026            .map(Party::new)
027            .collect(toSet());
028
029        this.addUnseenParties(initialParties);
030    }
031
032    @Override
033    public void addUnseenParties(final Collection<Party> partiesToRequestFrom) {
034        partiesToRequestFrom
035            .stream()
036            .map(party -> new SyncableParty(party, 0))
037            .forEach(outstandingParties::add);
038    }
039
040    @Override
041    public Optional<SyncableParty> getNextParty() {
042        return Optional.ofNullable(this.outstandingParties.poll());
043    }
044
045    @Override
046    public void incrementFailedAttempt(final SyncableParty attemptedParty) {
047
048        if (attemptedParty.getAttempts() < MAX_ATTEMPTS) {
049            final SyncableParty updatedParty = new SyncableParty(
050                attemptedParty.getParty(), attemptedParty.getAttempts() + 1
051            );
052
053            this.outstandingParties.add(updatedParty);
054        }
055
056    }
057
058}
