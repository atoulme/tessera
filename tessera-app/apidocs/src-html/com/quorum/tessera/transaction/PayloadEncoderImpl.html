001package com.quorum.tessera.transaction;
002
003import com.quorum.tessera.transaction.model.EncodedPayload;
004import com.quorum.tessera.transaction.model.EncodedPayloadWithRecipients;
005import com.quorum.tessera.util.BinaryEncoder;
006import com.quorum.tessera.nacl.Key;
007import com.quorum.tessera.nacl.Nonce;
008import java.nio.ByteBuffer;
009import java.util.ArrayList;
010import java.util.List;
011
012import static java.util.stream.Collectors.toList;
013
014public class PayloadEncoderImpl implements PayloadEncoder, BinaryEncoder {
015
016    @Override
017    public byte[] encode(final EncodedPayload encodedPayload) {
018        final byte[] senderKey = encodeField(encodedPayload.getSenderKey().getKeyBytes());
019        final byte[] cipherText = encodeField(encodedPayload.getCipherText());
020        final byte[] nonce = encodeField(encodedPayload.getCipherTextNonce().getNonceBytes());
021        final byte[] recipientNonce = encodeField(encodedPayload.getRecipientNonce().getNonceBytes());
022        final byte[] recipients = encodeArray(encodedPayload.getRecipientBoxes());
023
024        return ByteBuffer.allocate(senderKey.length + cipherText.length + nonce.length + recipients.length + recipientNonce.length)
025                .put(senderKey)
026                .put(cipherText)
027                .put(nonce)
028                .put(recipients)
029                .put(recipientNonce)
030                .array();
031    }
032
033    @Override
034    public EncodedPayload decode(final byte[] input) {
035        final ByteBuffer buffer = ByteBuffer.wrap(input);
036
037        final long senderSize = buffer.getLong();
038        final byte[] senderKey = new byte[Math.toIntExact(senderSize)];
039        buffer.get(senderKey);
040
041        final long cipherTextSize = buffer.getLong();
042        final byte[] cipherText = new byte[Math.toIntExact(cipherTextSize)];
043        buffer.get(cipherText);
044
045        final long nonceSize = buffer.getLong();
046        final byte[] nonce = new byte[Math.toIntExact(nonceSize)];
047        buffer.get(nonce);
048
049        final long numberOfRecipients = buffer.getLong();
050        final List<byte[]> recipientBoxes = new ArrayList<>();
051        for (long i = 0; i < numberOfRecipients; i++) {
052            final long boxSize = buffer.getLong();
053            final byte[] box = new byte[Math.toIntExact(boxSize)];
054            buffer.get(box);
055            recipientBoxes.add(box);
056        }
057
058        final long recipientNonceSize = buffer.getLong();
059        final byte[] recipientNonce = new byte[Math.toIntExact(recipientNonceSize)];
060        buffer.get(recipientNonce);
061
062        return new EncodedPayload(
063                new Key(senderKey),
064                cipherText,
065                new Nonce(nonce),
066                recipientBoxes,
067                new Nonce(recipientNonce)
068        );
069    }
070
071    @Override
072    public byte[] encode(final EncodedPayloadWithRecipients encodedPayloadWithRecipients) {
073        final byte[] payloadBytes = encode(encodedPayloadWithRecipients.getEncodedPayload());
074
075        final List<byte[]> keysAsBytes = encodedPayloadWithRecipients
076                .getRecipientKeys()
077                .stream()
078                .map(Key::getKeyBytes)
079                .collect(toList());
080
081        final byte[] recipientBytes = encodeArray(keysAsBytes);
082
083        final byte[] returnValue = new byte[payloadBytes.length + recipientBytes.length];
084
085        System.arraycopy(payloadBytes, 0, returnValue, 0, payloadBytes.length);
086        System.arraycopy(recipientBytes, 0, returnValue, payloadBytes.length, recipientBytes.length);
087
088        return returnValue;
089    }
090
091    @Override
092    public EncodedPayloadWithRecipients decodePayloadWithRecipients(final byte[] input) {
093        final ByteBuffer buffer = ByteBuffer.wrap(input);
094
095        final long senderSize = buffer.getLong();
096        final byte[] senderKey = new byte[Math.toIntExact(senderSize)];
097        buffer.get(senderKey);
098
099        final long cipherTextSize = buffer.getLong();
100        final byte[] cipherText = new byte[Math.toIntExact(cipherTextSize)];
101        buffer.get(cipherText);
102
103        final long nonceSize = buffer.getLong();
104        final byte[] nonce = new byte[Math.toIntExact(nonceSize)];
105        buffer.get(nonce);
106
107        final long numberOfRecipients = buffer.getLong();
108        final List<byte[]> recipientBoxes = new ArrayList<>();
109        for (long i = 0; i < numberOfRecipients; i++) {
110            final long boxSize = buffer.getLong();
111            final byte[] box = new byte[Math.toIntExact(boxSize)];
112            buffer.get(box);
113            recipientBoxes.add(box);
114        }
115
116        final long recipientNonceSize = buffer.getLong();
117        final byte[] recipientNonce = new byte[Math.toIntExact(recipientNonceSize)];
118        buffer.get(recipientNonce);
119
120        EncodedPayload payload = new EncodedPayload(
121                new Key(senderKey),
122                cipherText,
123                new Nonce(nonce),
124                recipientBoxes,
125                new Nonce(recipientNonce)
126        );
127
128        final long recipientLength = buffer.getLong();
129
130        final List<byte[]> recipientKeys = new ArrayList<>();
131        for (long i = 0; i < recipientLength; i++) {
132            final long boxSize = buffer.getLong();
133            final byte[] box = new byte[Math.toIntExact(boxSize)];
134            buffer.get(box);
135            recipientKeys.add(box);
136        }
137
138        return new EncodedPayloadWithRecipients(
139                payload,
140                recipientKeys.stream().map(Key::new).collect(toList())
141        );
142    }
143
144}
