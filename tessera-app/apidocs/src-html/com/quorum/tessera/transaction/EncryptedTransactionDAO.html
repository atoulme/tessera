001package com.quorum.tessera.transaction;
002
003import com.quorum.tessera.enclave.model.MessageHash;
004import com.quorum.tessera.transaction.model.EncryptedTransaction;
005
006import java.util.List;
007import java.util.Optional;
008
009/**
010 * A data store for transactions that need to be retrieved later
011 */
012public interface EncryptedTransactionDAO {
013
014    /**
015     * Save a new Encrypted Transaction
016     * All fields are required to be non-null on the entity
017     *
018     * @param entity The entity to be persisted
019     * @return The entity that was persisted
020     */
021    EncryptedTransaction save(EncryptedTransaction entity);
022
023    /**
024     * Retrieve a transaction based on its hash
025     *
026     * @param hash the hash of the transaction to retrieve
027     * @return the encrypted transaction with the given hash
028     */
029    Optional<EncryptedTransaction> retrieveByHash(MessageHash hash);
030
031    /**
032     * Retrieves a list of all transactions stored in the database
033     *
034     * @return The list of all rows in the database
035     */
036    List<EncryptedTransaction> retrieveAllTransactions();
037
038    /**
039     * Deletes a transaction that has the given hash as its digest
040     *
041     * @param hash The hash of the message to be deleted
042     * @throws javax.persistence.EntityNotFoundException 
043     */
044    void delete(MessageHash hash);
045
046}
