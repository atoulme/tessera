001package com.quorum.tessera.transaction;
002
003import com.quorum.tessera.enclave.model.MessageHash;
004import com.quorum.tessera.transaction.model.EncryptedTransaction;
005import org.slf4j.Logger;
006import org.slf4j.LoggerFactory;
007
008import javax.persistence.EntityManager;
009import javax.persistence.EntityNotFoundException;
010import javax.persistence.PersistenceContext;
011import java.util.Arrays;
012import java.util.List;
013import java.util.Optional;
014
015/**
016 * A JPA implementation of {@link EncryptedTransactionDAO}
017 */
018public class EncryptedTransactionDAOImpl implements EncryptedTransactionDAO {
019
020    private static final Logger LOGGER = LoggerFactory.getLogger(EncryptedTransactionDAOImpl.class);
021
022    private static final String FIND_HASH_EQUAL
023        = "SELECT et FROM EncryptedTransaction et WHERE et.hash.hashBytes = :hash";
024
025    private static final String FIND_ALL = "SELECT et FROM EncryptedTransaction et";
026
027    @PersistenceContext(unitName = "tessera")
028    private EntityManager entityManager;
029
030    @Override
031    public EncryptedTransaction save(final EncryptedTransaction entity) {
032        entityManager.persist(entity);
033
034        LOGGER.debug("Persisting entity with hash {} and payload {}",
035            entity.getHash(), Arrays.toString(entity.getEncodedPayload())
036        );
037
038        return entity;
039    }
040
041    @Override
042    public Optional<EncryptedTransaction> retrieveByHash(final MessageHash hash) {
043        LOGGER.info("Retrieving payload with hash {}", hash);
044
045        return entityManager
046            .createQuery(FIND_HASH_EQUAL, EncryptedTransaction.class)
047            .setParameter("hash", hash.getHashBytes())
048            .getResultStream()
049            .findAny();
050    }
051
052    @Override
053    public List<EncryptedTransaction> retrieveAllTransactions() {
054        LOGGER.info("Fetching all EncryptedTransaction database rows");
055
056        return entityManager
057            .createQuery(FIND_ALL, EncryptedTransaction.class)
058            .getResultList();
059    }
060
061
062    @Override
063    public void delete(final MessageHash hash) {
064        LOGGER.info("Deleting transaction with hash {}", hash);
065
066        final EncryptedTransaction message = entityManager
067            .createQuery(FIND_HASH_EQUAL, EncryptedTransaction.class)
068            .setParameter("hash", hash.getHashBytes())
069            .getResultStream()
070            .findAny()
071            .orElseThrow(EntityNotFoundException::new);
072
073        entityManager.remove(message);
074    }
075
076}
