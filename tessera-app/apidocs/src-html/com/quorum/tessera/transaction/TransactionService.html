001package com.quorum.tessera.transaction;
002
003import com.quorum.tessera.enclave.model.MessageHash;
004import com.quorum.tessera.nacl.Key;
005import com.quorum.tessera.transaction.exception.TransactionNotFoundException;
006import com.quorum.tessera.transaction.model.EncodedPayloadWithRecipients;
007
008import java.util.Collection;
009import java.util.List;
010
011public interface TransactionService {
012
013    /**
014     * Deletes a particular transaction from the enclave
015     * Returns whether the desired state was achieved
016     * i.e. {@code true} if the message no longer exists, {@code false} if the message still exists
017     * <p>
018     * Note that the method will return true if trying to delete a non-existant message,
019     * since the desired state of the message no longer existing is satisfied
020     *
021     * @param hash The hash of the payload that should be deleted
022     */
023    void delete(MessageHash hash);
024
025    /**
026     * Retrieves all payloads that the provided key is a recipient to
027     *
028     * @param recipientPublicKey The key for which to search over
029     * @return The list of all payloads that have the given key as a recipient
030     */
031    Collection<EncodedPayloadWithRecipients> retrieveAllForRecipient(Key recipientPublicKey);
032
033    /**
034     * Retrieves the message specified by the provided hash,
035     * and checks the intended recipient is present in the recipient list
036     * <p>
037     * If the given recipient is present, then modifies the returned payload to only include
038     * that recipient in the sealed box list so that it can be sent to the other nodes
039     *
040     * @param hash The hash of the message to retrieve
041     * @return The encrypted payload if found
042     * @throws NullPointerException         if the hash or intended recipient is null
043     * @throws TransactionNotFoundException if the transaction is not found
044     */
045    EncodedPayloadWithRecipients retrievePayload(MessageHash hash);
046
047    /**
048     * Retrieves the message specified by the provided hash,
049     * and uses the provided key as the sender of the message
050     * <p>
051     * Returns the unencrypted payload that is contained within so that
052     * Quorum can process it
053     *
054     * @param hash   The hash of the message to retrieve
055     * @param sender The public key of the sender to use (or {@code null} if we are the originator)
056     * @return The encrypted payload if the sender keys match
057     * @throws NullPointerException if the hash or sender is null
058     * @throws RuntimeException     if the provided sender is not the original sender of the message
059     */
060    byte[] retrieveUnencryptedTransaction(MessageHash hash, Key sender);
061
062    /**
063     * Store the sealed payload in the database that was propagated to from another node
064     *
065     * @param payloadWithRecipients the decoded message that gets sent between nodes
066     * @return the SHA3-513 hash of the encrypted message
067     */
068    MessageHash storeEncodedPayload(EncodedPayloadWithRecipients payloadWithRecipients);
069
070    /**
071     * Encrypts the payload using each of the given recipient keys and a random nonce
072     * against the given sender's public key.
073     * <p>
074     * Produces a result that maps the given recipient key and nonce value against the resulting ciphertext
075     *
076     * @param message             The message to be encrypted
077     * @param senderPublicKey     The public key of the sender
078     * @param recipientPublicKeys The list of public keys that are recipients of the this message
079     * @return A map of keys to nonces to cipher texts
080     */
081    EncodedPayloadWithRecipients encryptPayload(byte[] message, Key senderPublicKey, List<Key> recipientPublicKeys);
082
083}
