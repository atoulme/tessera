001package com.quorum.tessera.util;
002
003import com.quorum.tessera.util.exception.DecodingException;
004
005import java.util.Base64;
006
007/**
008 * An delegate interface for {@link Base64} to allow mocking in tests
009 */
010public interface Base64Decoder {
011
012    /**
013     * Decodes a Base64 encoded string to its constituent bytes
014     *
015     * @param data the input string that is Base64 encoded
016     * @return the decoded input
017     * @throws DecodingException if the input is not valid Base64
018     */
019    default byte[] decode(final String data) {
020        try {
021            return Base64.getDecoder().decode(data);
022        } catch (final IllegalArgumentException ex) {
023            throw new DecodingException(ex);
024        }
025    }
026
027    /**
028     * Encodes the given input to a Base64 encoded string
029     * Does not trim the end if the input was padded
030     *
031     * @param data the input data to encode to Base64
032     * @return the encoded form of the input
033     */
034    default String encodeToString(final byte[] data) {
035        return Base64.getEncoder().encodeToString(data);
036    }
037
038    static Base64Decoder create() {
039        return new Base64Decoder() {
040        };
041    }
042
043}
