001package com.quorum.tessera.key;
002
003import com.quorum.tessera.config.KeyData;
004import com.quorum.tessera.key.exception.KeyNotFoundException;
005import com.quorum.tessera.nacl.Key;
006import com.quorum.tessera.nacl.KeyPair;
007import org.slf4j.Logger;
008import org.slf4j.LoggerFactory;
009
010import java.util.*;
011import java.util.stream.Collectors;
012
013public class KeyManagerImpl implements KeyManager {
014
015    private static final Logger LOGGER = LoggerFactory.getLogger(KeyManagerImpl.class);
016
017    /**
018     * A list of all pub/priv keys that are attached to this node
019     */
020    private final Set<KeyPair> localKeys;
021
022    private final KeyPair defaultKeys;
023
024    private final Set<Key> forwardingPublicKeys;
025
026    public KeyManagerImpl(final Collection<KeyData> keys, Collection<Key> forwardKeys) {
027        this.localKeys = keys
028            .stream()
029            .map(kd ->
030                new KeyPair(
031                    new Key(Base64.getDecoder().decode(kd.getPublicKey())),
032                    new Key(Base64.getDecoder().decode(kd.getPrivateKey()))
033                )
034            ).collect(Collectors.toSet());
035
036        this.defaultKeys = localKeys.iterator().next();
037
038        this.forwardingPublicKeys = Collections.unmodifiableSet(new HashSet<>(forwardKeys));
039    }
040
041    @Override
042    public Key getPublicKeyForPrivateKey(final Key privateKey) {
043        LOGGER.debug("Attempting to find public key for the private key {}", privateKey);
044
045        final Key publicKey = localKeys
046            .stream()
047            .filter(keypair -> Objects.equals(keypair.getPrivateKey(), privateKey))
048            .findFirst()
049            .map(KeyPair::getPublicKey)
050            .orElseThrow(
051                () -> new KeyNotFoundException("Private key " + privateKey + " not found when searching for public key")
052            );
053
054        LOGGER.debug("Found public key {} for private key {}", publicKey, privateKey);
055
056        return publicKey;
057    }
058
059    @Override
060    public Key getPrivateKeyForPublicKey(final Key publicKey) {
061        LOGGER.debug("Attempting to find private key for the public key {}", publicKey);
062
063        final Key privateKey = localKeys
064            .stream()
065            .filter(keypair -> Objects.equals(keypair.getPublicKey(), publicKey))
066            .findFirst()
067            .map(KeyPair::getPrivateKey)
068            .orElseThrow(
069                () -> new KeyNotFoundException("Public key " + publicKey + " not found when searching for private key")
070            );
071
072        LOGGER.debug("Found private key {} for public key {}", privateKey, publicKey);
073
074        return privateKey;
075    }
076
077    @Override
078    public Set<Key> getPublicKeys() {
079        return localKeys
080            .stream()
081            .map(KeyPair::getPublicKey)
082            .collect(Collectors.toSet());
083    }
084
085    @Override
086    public Key defaultPublicKey() {
087        return defaultKeys.getPublicKey();
088    }
089
090    @Override
091    public Set<Key> getForwardingKeys() {
092        return this.forwardingPublicKeys;
093    }
094
095}
