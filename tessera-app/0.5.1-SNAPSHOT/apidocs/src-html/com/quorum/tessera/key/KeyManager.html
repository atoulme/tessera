001package com.quorum.tessera.key;
002
003import com.quorum.tessera.nacl.Key;
004
005import java.util.Set;
006
007/**
008 * Manages local keys for the running node, include key lookups and key loading/generation
009 */
010public interface KeyManager {
011
012    /**
013     * Fetches the public key that corresponds to the given private key
014     * This means that the returned public key is under control of the current node
015     * (i.e. it is one of our keys)
016     *
017     * @param privateKey the private key for which to find the public key
018     * @return the corresponding public key
019     * @throws RuntimeException if the private key could not be found
020     */
021    Key getPublicKeyForPrivateKey(Key privateKey);
022
023    /**
024     * Fetches the private key that corresponds to the given public key
025     * This means that the returned private key is under control of the current node
026     * (i.e. it is one of our keys)
027     *
028     * @param publicKey the public key for which to find the private key
029     * @return the corresponding private key
030     * @throws RuntimeException if the public key could not be found
031     */
032    Key getPrivateKeyForPublicKey(Key publicKey);
033
034    /**
035     * Return a list of all recipients public keys of this node
036     *
037     * @return the set of all public keys
038     */
039    Set<Key> getPublicKeys();
040
041    /**
042     * Returns the default public key of this node if a transaction is
043     * sent that doesn't have a sender.
044     * <p>
045     * Must return the same key each time during a single run,
046     * but no guarantee is made about which key is returned if the
047     * application is restarted.
048     *
049     * @return The public key to use as the default sender
050     */
051    Key defaultPublicKey();
052
053    /**
054     * Returns a set of public keys that all transactions should be sent to in
055     * addition to the declared recipients
056     *
057     * @return the set of keys to forward to
058     */
059    Set<Key> getForwardingKeys();
060
061}
