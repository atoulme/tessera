001package com.quorum.tessera.api;
002
003import io.swagger.annotations.ApiResponse;
004import io.swagger.annotations.ApiResponses;
005import io.swagger.annotations.ApiOperation;
006
007import javax.ws.rs.GET;
008import javax.ws.rs.Path;
009import javax.ws.rs.Produces;
010import javax.ws.rs.core.MediaType;
011
012import org.slf4j.Logger;
013import org.slf4j.LoggerFactory;
014
015/**
016 * Provides endpoints to determine versioning information
017 */
018@Path("/version")
019public class VersionResource {
020
021    private static final Logger LOGGER = LoggerFactory.getLogger(VersionResource.class);
022
023    private static final String VERSION = "No version defined yet!";
024
025    /**
026     * An endpoint describing the current version of the application
027     *
028     * @return the version of the application
029     */
030    @GET
031    @Produces(MediaType.TEXT_PLAIN)
032    @ApiOperation(value = "Request current version of Tessera", produces = "current version number")
033    @ApiResponses({@ApiResponse(code = 200, message = "Current application version ", response = String.class)})
034    public String getVersion() {
035
036        LOGGER.info("GET version");
037
038        return VERSION;
039    }
040}
