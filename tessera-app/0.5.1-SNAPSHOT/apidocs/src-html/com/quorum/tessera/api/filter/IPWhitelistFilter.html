001package com.quorum.tessera.api.filter;
002
003import com.quorum.tessera.config.Config;
004import com.quorum.tessera.config.Peer;
005import javax.servlet.http.HttpServletRequest;
006import javax.ws.rs.container.ContainerRequestContext;
007import javax.ws.rs.container.ContainerRequestFilter;
008import javax.ws.rs.core.Context;
009import javax.ws.rs.core.Response;
010import java.util.Set;
011import java.util.stream.Collectors;
012
013/**
014 * Applies a filter to all endpoints that only allows certain IP address and ghost names to
015 * get access to the HTTP endpoints
016 *
017 * If an error occurs whilst checking the whitelist,
018 * the filter is disabled. This is done since not all webservers have support for
019 * the {@link HttpServletRequest} context class, which is required.
020 */
021@GlobalFilter
022public class IPWhitelistFilter implements ContainerRequestFilter {
023
024    private final Set<String> whitelisted;
025
026    private boolean disabled;
027
028    private HttpServletRequest httpServletRequest;
029
030    public IPWhitelistFilter(final Config configuration) {
031        this.whitelisted = configuration.getPeers().stream()
032                .map(Peer::getUrl).collect(Collectors.toSet());
033        this.disabled = !configuration.isUseWhiteList();
034    }
035
036    /**
037     * If the filter is disabled, return immediately
038     * Otherwise, extract the callers hostname and address, and check it against the whitelist
039     *
040     * If a problem occurs, then disable the filter
041     *
042     * If the host is not whitelisted, finish the filter chain here and return an Unauthorized response
043     *
044     * @param requestContext the context of the current request
045     */
046    @Override
047    public void filter(final ContainerRequestContext requestContext) {
048
049        if(disabled) {
050            return;
051        }
052
053        try {
054
055            final String remoteAddress = httpServletRequest.getRemoteAddr();
056            final String remoteHost = httpServletRequest.getRemoteHost();
057
058            final boolean allowed = whitelisted.contains(remoteAddress) || whitelisted.contains(remoteHost);
059
060            if (!allowed) {
061                requestContext.abortWith(Response.status(Response.Status.UNAUTHORIZED).build());
062            }
063
064        } catch (final Exception ex) {
065            this.disabled = true;
066        }
067
068    }
069
070    /**
071     * Apply the current HTTP context to the filter, to check the remote host
072     *
073     * @param request the request to be filtered
074     */
075    @Context
076    public void setHttpServletRequest(final HttpServletRequest request) {
077        this.httpServletRequest = request;
078    }
079
080}
