001package com.quorum.tessera.transaction;
002
003import com.quorum.tessera.enclave.model.MessageHash;
004import com.quorum.tessera.key.KeyManager;
005import com.quorum.tessera.nacl.Key;
006import com.quorum.tessera.nacl.NaclFacade;
007import com.quorum.tessera.nacl.Nonce;
008import com.quorum.tessera.transaction.exception.TransactionNotFoundException;
009import com.quorum.tessera.transaction.model.EncodedPayload;
010import com.quorum.tessera.transaction.model.EncodedPayloadWithRecipients;
011import com.quorum.tessera.transaction.model.EncryptedTransaction;
012import org.bouncycastle.jcajce.provider.digest.SHA3;
013import org.slf4j.Logger;
014import org.slf4j.LoggerFactory;
015
016import javax.transaction.Transactional;
017import java.util.Collection;
018import java.util.List;
019import java.util.Objects;
020import java.util.stream.Collectors;
021
022import static java.util.stream.Collectors.toList;
023
024@Transactional
025public class TransactionServiceImpl implements TransactionService {
026
027    private static final Logger LOGGER = LoggerFactory.getLogger(TransactionServiceImpl.class);
028
029    private final EncryptedTransactionDAO encryptedTransactionDAO;
030
031    private final PayloadEncoder payloadEncoder;
032
033    private final KeyManager keyManager;
034
035    private final NaclFacade nacl;
036
037    public TransactionServiceImpl(final EncryptedTransactionDAO encryptedTransactionDAO,
038                                  final PayloadEncoder payloadEncoder,
039                                  final KeyManager keyManager,
040                                  final NaclFacade nacl) {
041        this.encryptedTransactionDAO = Objects.requireNonNull(encryptedTransactionDAO);
042        this.payloadEncoder = Objects.requireNonNull(payloadEncoder);
043        this.keyManager = Objects.requireNonNull(keyManager);
044        this.nacl = Objects.requireNonNull(nacl);
045    }
046
047    @Override
048    public void delete(final MessageHash hash) {
049        LOGGER.info("Received request to delete message with hash {}", hash);
050        this.encryptedTransactionDAO.delete(hash);
051    }
052
053    @Override
054    public Collection<EncodedPayloadWithRecipients> retrieveAllForRecipient(final Key recipientPublicKey) {
055        LOGGER.debug("Retrieving all transaction for recipient {}", recipientPublicKey);
056
057        return encryptedTransactionDAO
058            .retrieveAllTransactions()
059            .stream()
060            .map(EncryptedTransaction::getEncodedPayload)
061            .map(payloadEncoder::decodePayloadWithRecipients)
062            .filter(payload -> payload.getRecipientKeys().contains(recipientPublicKey))
063            .collect(toList());
064    }
065
066    @Override
067    public EncodedPayloadWithRecipients retrievePayload(final MessageHash hash) {
068        return encryptedTransactionDAO
069            .retrieveByHash(hash)
070            .map(EncryptedTransaction::getEncodedPayload)
071            .map(payloadEncoder::decodePayloadWithRecipients)
072            .orElseThrow(() -> new TransactionNotFoundException("Message with hash " + hash + " was not found"));
073    }
074
075    @Override
076    public byte[] retrieveUnencryptedTransaction(final MessageHash hash, final Key providedKey) {
077
078        final EncryptedTransaction encryptedTransaction = encryptedTransactionDAO
079            .retrieveByHash(hash)
080            .orElseThrow(() -> new TransactionNotFoundException("Message with hash " + hash + " was not found"));
081
082        final EncodedPayloadWithRecipients payloadWithRecipients
083            = payloadEncoder.decodePayloadWithRecipients(encryptedTransaction.getEncodedPayload());
084
085        final EncodedPayload encodedPayload = payloadWithRecipients.getEncodedPayload();
086
087        final Key senderPubKey;
088       
089        final Key recipientPubKey;
090
091        if (!keyManager.getPublicKeys().contains(encodedPayload.getSenderKey())) {
092            // This is a payload originally sent to us by another node
093            recipientPubKey = encodedPayload.getSenderKey();
094            senderPubKey = providedKey;
095        } else {
096            // This is a payload that originated from us
097            senderPubKey = encodedPayload.getSenderKey();
098            recipientPubKey = payloadWithRecipients.getRecipientKeys().get(0);
099        }
100
101        final Key senderPrivKey = keyManager.getPrivateKeyForPublicKey(senderPubKey);
102        final Key sharedKey = nacl.computeSharedKey(recipientPubKey, senderPrivKey);
103
104        try {
105            final byte[] recipientBox = encodedPayload.getRecipientBoxes().iterator().next();
106            final Nonce nonce =  encodedPayload.getRecipientNonce();
107            final byte[] masterKeyBytes = nacl.openAfterPrecomputation(recipientBox,nonce, sharedKey);
108
109            final Key masterKey = new Key(masterKeyBytes);
110
111            final byte[] cipherText = encodedPayload.getCipherText();
112            final Nonce cipherTextNonce = encodedPayload.getCipherTextNonce();
113
114            return nacl.openAfterPrecomputation(cipherText, cipherTextNonce, masterKey);
115
116        } catch (final RuntimeException ex) {
117            LOGGER.info("Couldn't decrypt message with hash {}. Our public key is: {}", hash, senderPubKey);
118            LOGGER.debug("RuntimeException: ", ex);
119            throw ex;
120        }
121
122    }
123
124    @Override
125    public MessageHash storeEncodedPayload(final EncodedPayloadWithRecipients payloadWithRecipients) {
126
127        final SHA3.DigestSHA3 digestSHA3 = new SHA3.Digest512();
128        final byte[] digest = digestSHA3.digest(payloadWithRecipients.getEncodedPayload().getCipherText());
129
130        final MessageHash transactionHash = new MessageHash(digest);
131
132        LOGGER.info("Generated transaction hash {}", transactionHash);
133
134        final EncryptedTransaction newTransaction = new EncryptedTransaction(
135            transactionHash,
136            this.payloadEncoder.encode(payloadWithRecipients)
137        );
138
139        this.encryptedTransactionDAO.save(newTransaction);
140
141        return transactionHash;
142    }
143
144    @Override
145    public EncodedPayloadWithRecipients encryptPayload(final byte[] message,
146                                                       final Key senderPublicKey,
147                                                       final List<Key> recipientPublicKeys) {
148
149        final Key masterKey = nacl.createSingleKey();
150        final Nonce nonce = nacl.randomNonce();
151        final Nonce recipientNonce = nacl.randomNonce();
152
153        final byte[] cipherText = nacl.sealAfterPrecomputation(message, nonce, masterKey);
154
155        final Key privateKey = keyManager.getPrivateKeyForPublicKey(senderPublicKey);
156
157        final List<byte[]> encryptedMasterKeys = recipientPublicKeys
158            .stream()
159            .map(key -> nacl.computeSharedKey(key, privateKey))
160            .map(key -> nacl.sealAfterPrecomputation(masterKey.getKeyBytes(), recipientNonce, key))
161            .collect(Collectors.toList());
162
163        return new EncodedPayloadWithRecipients(
164            new EncodedPayload(senderPublicKey, cipherText, nonce, encryptedMasterKeys, recipientNonce),
165            recipientPublicKeys
166        );
167
168    }
169
170}
