001package com.quorum.tessera.sync;
002
003import com.quorum.tessera.api.model.ResendRequest;
004import com.quorum.tessera.api.model.ResendRequestType;
005import com.quorum.tessera.key.KeyManager;
006import com.quorum.tessera.nacl.Key;
007import com.quorum.tessera.node.PostDelegate;
008import org.slf4j.Logger;
009import org.slf4j.LoggerFactory;
010
011import java.util.Objects;
012
013public class TransactionRequesterImpl implements TransactionRequester {
014
015    private static final Logger LOGGER = LoggerFactory.getLogger(TransactionRequesterImpl.class);
016
017    private final KeyManager keyManager;
018
019    private PostDelegate client;
020
021    public TransactionRequesterImpl(final KeyManager keyManager,
022                                    final PostDelegate client) {
023        this.keyManager = Objects.requireNonNull(keyManager);
024        this.client = Objects.requireNonNull(client);
025    }
026
027    @Override
028    public boolean requestAllTransactionsFromNode(final String uri) {
029
030        LOGGER.debug("Requesting transactions get resent for {}", uri);
031
032        return this.keyManager
033            .getPublicKeys()
034            .parallelStream()
035            .map(this::createRequestAllEntity)
036            .allMatch(req -> this.makeRequest(uri, req));
037
038    }
039
040    /**
041     * Will make the desired request until succeeds or max tries has been reached
042     *
043     * @param uri     the URI to call
044     * @param request the request object to send
045     */
046    private boolean makeRequest(final String uri, final ResendRequest request) {
047        LOGGER.debug("Requesting a resend for key {}", request.getPublicKey());
048
049        boolean success;
050        int numberOfTries = 0;
051
052        do {
053            try {
054                success = this.client.makeResendRequest(uri, request);
055            } catch (final Exception ex) {
056                success = false;
057                LOGGER.debug("Failed to make resend request to node {} for key {}", uri, request.getPublicKey());
058            }
059
060            numberOfTries++;
061
062        } while (!success && (numberOfTries < MAX_ATTEMPTS));
063
064        return success;
065
066    }
067
068    /**
069     * Creates the entity that should be sent to the target URL
070     *
071     * @param key the public key that transactions should be resent for
072     * @return the request to be sent
073     */
074    private ResendRequest createRequestAllEntity(final Key key) {
075
076        final ResendRequest request = new ResendRequest();
077        request.setPublicKey(key.toString());
078        request.setType(ResendRequestType.ALL);
079
080        return request;
081    }
082
083}
