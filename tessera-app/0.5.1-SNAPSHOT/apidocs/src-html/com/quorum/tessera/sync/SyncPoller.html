001package com.quorum.tessera.sync;
002
003import com.quorum.tessera.sync.model.SyncableParty;
004
005import java.util.Objects;
006import java.util.Optional;
007import java.util.concurrent.ExecutorService;
008
009/**
010 * A poller that will contact all outstanding parties that need to have
011 * transactions resent for a single round
012 */
013public class SyncPoller implements Runnable {
014
015    private final ExecutorService executorService;
016
017    private final ResendPartyStore resendPartyStore;
018
019    private final TransactionRequester transactionRequester;
020
021    public SyncPoller(final ExecutorService executorService,
022                      final ResendPartyStore resendPartyStore,
023                      final TransactionRequester transactionRequester) {
024        this.executorService = Objects.requireNonNull(executorService);
025        this.resendPartyStore = Objects.requireNonNull(resendPartyStore);
026        this.transactionRequester = Objects.requireNonNull(transactionRequester);
027    }
028
029    /**
030     * Retrieves all of the outstanding parties and makes an attempt to make the resend request
031     * asynchronously. If the request fails then the party is submitted back to the store for
032     * a later attempt.
033     */
034    @Override
035    public void run() {
036
037        Optional<SyncableParty> nextPartyToSend = this.resendPartyStore.getNextParty();
038
039        while (nextPartyToSend.isPresent()) {
040
041            final SyncableParty requestDetails = nextPartyToSend.get();
042            final String url = requestDetails.getParty().getUrl();
043
044            final Runnable action = () -> {
045                final boolean allSucceeded = this.transactionRequester.requestAllTransactionsFromNode(url);
046
047                if (!allSucceeded) {
048                    this.resendPartyStore.incrementFailedAttempt(requestDetails);
049                }
050            };
051
052            this.executorService.submit(action);
053
054            nextPartyToSend = this.resendPartyStore.getNextParty();
055
056        }
057
058    }
059
060}
