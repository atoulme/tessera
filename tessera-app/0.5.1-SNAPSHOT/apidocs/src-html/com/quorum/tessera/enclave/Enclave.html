001package com.quorum.tessera.enclave;
002
003import com.quorum.tessera.enclave.model.MessageHash;
004import com.quorum.tessera.nacl.Key;
005import com.quorum.tessera.transaction.model.EncodedPayloadWithRecipients;
006
007import java.util.Optional;
008
009public interface Enclave {
010
011    /**
012     * Deletes a particular transaction from the enclave
013     * Returns whether the desired state was achieved
014     * i.e. {@code true} if the message no longer exists, {@code false} if the message still exists
015     * <p>
016     * Note that the method will return true if trying to delete a non-existant message,
017     * since the desired state of the message no longer existing is satisfied
018     *
019     * @param hashBytes The hash of the payload that should be deleted
020     */
021    void delete(byte[] hashBytes);
022
023    /**
024     * Retrieves the message specified by the provided hash,
025     * and uses the provided key. If key is not provided,
026     * method will use the default public key, which is public key of the current node
027     * @param key The hash of payload to retrieve
028     * @param to
029     * @return
030     */
031    byte[] receive(byte[] key, Optional<byte[]> to);
032
033    /**
034     * Create an EncodedPayloadWithRecipients, then store this into the database,
035     * also publish the the payload to all recipients
036     * If sender public key is not provided, method will use default public key,
037     * which is public key of the current node
038     * @param sender Public key of the sender
039     * @param recipients List of recipients for this transaction
040     * @param message
041     * @return The message hash returned after the transaction is  persisted.
042     * This message hash can be used to retrieve the transaction.
043     */
044    MessageHash store(Optional<byte[]> sender, byte[][] recipients, byte[] message);
045
046    /**
047     * Decode raw payload from byte array to EncodedPayloadWithRecipients,
048     * and persist to the database
049     * @param encodedPayloadWithRecipients
050     * @return The message hash returned after the transaction is  persisted.
051     * This message hash can be used to retrieve the transaction
052     */
053    MessageHash storePayload(byte[] encodedPayloadWithRecipients);
054
055    /**
056     * Publish the encoded payload to the specified recipient.
057     * This method will strip out irrelevant recipients and recipient boxes from the payload
058     * before sending. If the recipient
059     * @param encodedPayload Payload to send to recipient specified
060     * @param recipient public key of the recipient to send payload to
061     */
062    void publishPayload(EncodedPayloadWithRecipients encodedPayload, Key recipient);
063
064    /**
065     * Retrieves all payloads that the provided key is a recipient to,
066     * then publish the payload to all recipients in the recipient list of the payload
067     * @param recipientKey The key for which to search over
068     */
069    void resendAll(byte[] recipientKey);
070
071    /**
072     * Fetch a transaction with the given hash
073     * The transaction should have originated from this node and the
074     * given key should appear in the recipients list
075     *
076     * @param hash the hash of the transaction to find
077     * @param recipient the recipient of the transaction
078     * @return The payload that was originally distributed to the node that managed the recipient
079     */
080    EncodedPayloadWithRecipients fetchTransactionForRecipient(MessageHash hash, Key recipient);
081}
