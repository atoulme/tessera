001package com.quorum.tessera.nacl.jnacl;
002
003import com.quorum.tessera.nacl.NaclFacade;
004import com.quorum.tessera.nacl.NaclFacadeFactory;
005import org.slf4j.Logger;
006import org.slf4j.LoggerFactory;
007
008import java.security.SecureRandom;
009
010/**
011 * Provides the JNaCL implementation of the {@link NaclFacade}
012 */
013public class JnaclFactory implements NaclFacadeFactory {
014
015    private static final Logger LOGGER = LoggerFactory.getLogger(JnaclFactory.class);
016
017    @Override
018    public NaclFacade create() {
019        LOGGER.debug("Creating a JNaCL implementation of NaclFacadeFactory");
020
021        final SecureRandom secureRandom = new SecureRandom();
022        final JnaclSecretBox secretBox = new JnaclSecretBox();
023
024        return new Jnacl(secureRandom, secretBox);
025    }
026
027}
