001package com.quorum.tessera.nacl.jnacl;
002
003import com.neilalexander.jnacl.crypto.curve25519xsalsa20poly1305;
004
005/**
006 * An interface to make testing of the static methods provided by
007 * {@link curve25519xsalsa20poly1305} easier.
008 */
009public interface SecretBox {
010
011    /**
012     * Computes a shared key between the given public and private key
013     *
014     * @param output The output array that contains the shared key
015     * @param publicKey The bytes of the public key
016     * @param privateKey The bytes of the private key
017     * @return Whether the operation was successful (0 for success, -1 for failure)
018     */
019    int cryptoBoxBeforenm(byte[] output, byte[] publicKey, byte[] privateKey);
020
021    /**
022     * Seal a message into ciphertext using a shared key
023     *
024     * @param output the output ciphertext that contains the required 0 padding from NaCL
025     * @param message the input message with the require 0 padding from NaCL
026     * @param messageLength the length of the padding input message
027     * @param nonce a unique nonce to encrypt the message with
028     * @param sharedKey the shared key to encrypt the message with
029     * @return Whether the operation was successful (0 for success, -1 for failure)
030     */
031    int cryptoBoxAfternm(byte[] output, byte[] message, int messageLength, byte[] nonce, byte[] sharedKey);
032
033    /**
034     * Opens a sealed box using the provided shared key
035     *
036     * @param output the output array for the unencrpyted message
037     * @param message the cipher text with the require 0 padding from NaCL
038     * @param messageLength the length of the message + the required 0 padding from NaCL
039     * @param nonce the nonce that was used to seal the box
040     * @param sharedKey the shared key
041     * @return Whether the operation was successful (0 for success, -1 for failure)
042     */
043    int cryptoBoxOpenAfternm(byte[] output, byte[] message, int messageLength, byte[] nonce, byte[] sharedKey);
044
045    /**
046     * Generates a new keypair
047     * The input arrays must be at least 32 bytes long (and only the first 32 bytes are populated
048     * if the array is longer)
049     *
050     * @param publicKey The output array for the public key
051     * @param privateKey The output array for the private key
052     * @return Whether the operation was successful (0 for success, -1 for failure)
053     */
054    int cryptoBoxKeypair(byte[] publicKey, byte[] privateKey);
055
056}
