001package com.quorum.tessera.nacl.jnacl;
002
003import com.neilalexander.jnacl.NaCl;
004import com.quorum.tessera.nacl.*;
005import org.slf4j.Logger;
006import org.slf4j.LoggerFactory;
007
008import java.security.SecureRandom;
009import java.util.Arrays;
010import java.util.Objects;
011
012import static com.neilalexander.jnacl.crypto.curve25519xsalsa20poly1305.*;
013
014/**
015 * Uses JNaCL, which is a pure Java implementation of the NaCL standard
016 */
017public class Jnacl implements NaclFacade {
018
019    private static final Logger LOGGER = LoggerFactory.getLogger(Jnacl.class);
020
021    private static final String REDACTED = "REDACTED";
022
023    private final SecureRandom secureRandom;
024
025    private final SecretBox secretBox;
026
027    public Jnacl(final SecureRandom secureRandom, final SecretBox secretBox) {
028        this.secureRandom = Objects.requireNonNull(secureRandom);
029        this.secretBox = Objects.requireNonNull(secretBox);
030    }
031
032    @Override
033    public Key computeSharedKey(final Key publicKey, final Key privateKey) {
034        final byte[] precomputed = new byte[crypto_secretbox_BEFORENMBYTES];
035
036        LOGGER.debug("Computing the shared key for public key {} and private key {}", publicKey, privateKey);
037        final int jnaclResult = secretBox.cryptoBoxBeforenm(
038            precomputed, publicKey.getKeyBytes(), privateKey.getKeyBytes()
039        );
040
041        if(jnaclResult == -1) {
042            LOGGER.warn("Could not compute the shared key for pub {} and priv {}", publicKey, REDACTED);
043            LOGGER.debug("Could not compute the shared key for pub {} and priv {}", publicKey, privateKey);
044            throw new NaclException("JNacl could not compute the shared key");
045        }
046
047        final Key sharedKey = new Key(precomputed);
048
049        LOGGER.debug("Computed shared key {} for pub {} and priv {}", sharedKey, publicKey, privateKey);
050
051        return sharedKey;
052
053    }
054
055    @Override
056    public byte[] seal(final byte[] message, final Nonce nonce, final Key publicKey, final Key privateKey) {
057
058        LOGGER.debug("Sealing message using public key {}", publicKey);
059        LOGGER.debug(
060            "Sealing message {} using nonce {}, public key {} and private key {}",
061            Arrays.toString(message), nonce, publicKey, privateKey
062        );
063
064        try {
065
066            final NaCl nacl = new NaCl(privateKey.getKeyBytes(), publicKey.getKeyBytes());
067
068            final byte[] cipherText = nacl.encrypt(message, nonce.getNonceBytes());
069
070            LOGGER.debug("Created sealed payload for public key {}", publicKey);
071            LOGGER.debug(
072                "Created sealed payload {} using nonce {}, public key {} and private key {}",
073                Arrays.toString(cipherText), nonce, publicKey, privateKey
074            );
075
076            return extract(cipherText, crypto_secretbox_BOXZEROBYTES);
077
078        } catch (final Exception ex) {
079            throw new NaclException(ex.getMessage());
080        }
081
082    }
083
084    @Override
085    public byte[] open(final byte[] cipherText, final Nonce nonce, final Key publicKey, final Key privateKey) {
086        LOGGER.debug("Opening message using public key {}", publicKey);
087        LOGGER.debug(
088            "Opening message {} using nonce {}, public key {} and private key {}",
089            Arrays.toString(cipherText), nonce, publicKey, privateKey
090        );
091
092        try {
093
094            final byte[] paddedInput = pad(cipherText, crypto_secretbox_BOXZEROBYTES);
095
096            final NaCl nacl = new NaCl(privateKey.getKeyBytes(), publicKey.getKeyBytes());
097
098            final byte[] plaintext = nacl.decrypt(paddedInput, nonce.getNonceBytes());
099
100            LOGGER.debug("Created sealed payload for public key {}", publicKey);
101            LOGGER.debug(
102                "Created sealed payload {} using nonce {}, public key {} and private key {}",
103                Arrays.toString(cipherText), nonce, publicKey, privateKey
104            );
105
106            return plaintext;
107        } catch (final Exception ex) {
108            throw new NaclException(ex.getMessage());
109        }
110    }
111
112    @Override
113    public byte[] sealAfterPrecomputation(final byte[] message, final Nonce nonce, final Key sharedKey) {
114
115        final byte[] paddedMessage = new byte[message.length + crypto_secretbox_ZEROBYTES];
116        final byte[] output = new byte[message.length + crypto_secretbox_ZEROBYTES];
117
118        LOGGER.debug("Sealing message using public key {}", sharedKey);
119        LOGGER.debug(
120            "Sealing message {} using nonce {} and shared key {}",
121            Arrays.toString(message), nonce, sharedKey
122        );
123
124        System.arraycopy(message, 0, paddedMessage, crypto_secretbox_ZEROBYTES, message.length);
125        final int jnaclResult = secretBox.cryptoBoxAfternm(
126            output, paddedMessage, paddedMessage.length, nonce.getNonceBytes(), sharedKey.getKeyBytes()
127        );
128
129        if(jnaclResult == -1) {
130            LOGGER.warn("Could not create sealed payload using shared key {}", sharedKey);
131            LOGGER.debug("Could not create sealed payload using shared key {}", sharedKey);
132            throw new NaclException("Kalium could not seal the payload using the shared key");
133        }
134
135        LOGGER.debug("Created sealed payload for shared key {}", sharedKey);
136        LOGGER.debug(
137            "Created sealed payload {} using nonce {} and shared key {}",
138            Arrays.toString(output), nonce, sharedKey
139        );
140
141        return extract(output, crypto_secretbox_BOXZEROBYTES);
142    }
143
144    @Override
145    public byte[] openAfterPrecomputation(final byte[] cipherText, final Nonce nonce, final Key sharedKey) {
146        LOGGER.debug("Opening message using shared key {}", sharedKey);
147        LOGGER.debug(
148            "Opening message {} using nonce {} and shared key {}",
149            Arrays.toString(cipherText), nonce, sharedKey
150        );
151
152        final byte[] paddedInput = pad(cipherText, crypto_secretbox_BOXZEROBYTES);
153        final byte[] paddedOutput = new byte[paddedInput.length];
154
155        final int jnaclResult = secretBox.cryptoBoxOpenAfternm(
156            paddedOutput, paddedInput, paddedInput.length, nonce.getNonceBytes(), sharedKey.getKeyBytes()
157        );
158
159        if(jnaclResult == -1) {
160            LOGGER.warn("Could not open sealed payload using shared key {}", sharedKey);
161            LOGGER.debug("Could not open sealed payload using shared key {}", sharedKey);
162            throw new NaclException("Kalium could not open the payload using the shared key");
163        }
164
165        LOGGER.debug("Opened sealed payload for shared key {}", sharedKey);
166        LOGGER.debug(
167            "Opened payload {} using nonce {}, public key {} and private key {} to get result {}",
168            Arrays.toString(cipherText), nonce, sharedKey, REDACTED, Arrays.toString(paddedOutput)
169        );
170
171        return extract(paddedOutput, crypto_secretbox_ZEROBYTES);
172    }
173
174    @Override
175    public Nonce randomNonce() {
176        final byte[] nonceBytes = new byte[crypto_secretbox_NONCEBYTES];
177
178        this.secureRandom.nextBytes(nonceBytes);
179
180        final Nonce nonce = new Nonce(nonceBytes);
181
182        LOGGER.debug("Generated random nonce {}", nonce);
183
184        return nonce;
185    }
186
187    @Override
188    public KeyPair generateNewKeys() {
189        final byte[] publicKey = new byte[crypto_secretbox_PUBLICKEYBYTES];
190        final byte[] privateKey = new byte[crypto_secretbox_SECRETKEYBYTES];
191
192        LOGGER.info("Generating new keypair...");
193
194        final int jnaclResult = secretBox.cryptoBoxKeypair(publicKey, privateKey);
195
196        if(jnaclResult == -1) {
197            LOGGER.warn("Unable to generate a new keypair!");
198            throw new NaclException("JNaCL could not generate a new public/private keypair");
199        }
200
201        final Key pubKey = new Key(publicKey);
202        final Key privKey = new Key(privateKey);
203
204        LOGGER.info("Generated public key {} and private key {}", pubKey, REDACTED);
205        LOGGER.debug("Generated public key {} and private key {}", pubKey, privKey);
206
207        return new KeyPair(pubKey, privKey);
208    }
209
210    @Override
211    public Key createSingleKey() {
212        LOGGER.debug("Generating random key");
213
214        final byte[] keyBytes = new byte[crypto_secretbox_PUBLICKEYBYTES];
215
216        this.secureRandom.nextBytes(keyBytes);
217
218        final Key key = new Key(keyBytes);
219
220        LOGGER.debug("Random key generated");
221        LOGGER.debug("Generated key with value {}", key);
222
223        return key;
224    }
225
226    /**
227     * Left-pads a given message with padSize amount of zeros
228     *
229     * @param input the message to be padded
230     * @param padSize the amount of left-padding to apply
231     * @return the padded message
232     */
233    private byte[] pad(final byte[] input, final int padSize) {
234        final byte[] paddedMessage = new byte[padSize + input.length];
235        System.arraycopy(input, 0, paddedMessage, padSize, input.length);
236
237        return paddedMessage;
238    }
239
240    /**
241     * Removes left-padding from a given message to tune of padSize
242     *
243     * @param input The message from which to remove left-padding
244     * @param padSize The amount of left-padding to remove
245     * @return The trimmed message
246     */
247    private byte[] extract(final byte[] input, final int padSize) {
248        final byte[] extractedMessage = new byte[input.length - padSize];
249        System.arraycopy(input, padSize, extractedMessage, 0, extractedMessage.length);
250
251        return extractedMessage;
252    }
