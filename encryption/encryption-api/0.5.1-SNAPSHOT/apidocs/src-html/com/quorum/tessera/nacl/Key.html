001package com.quorum.tessera.nacl;
002
003import java.util.Arrays;
004import java.util.Base64;
005
006/**
007 * Represents a Key, which is usually 32 bytes in length The possible types of
008 * keys include: - public - private - symmetric
009 */
010public class Key {
011
012    private final byte[] keyBytes;
013
014    public Key(final byte[] keyBytes) {
015        this.keyBytes = Arrays.copyOf(keyBytes, keyBytes.length);
016    }
017    
018    
019    public byte[] getKeyBytes() {
020        return Arrays.copyOf(this.keyBytes, this.keyBytes.length);
021    }
022
023    @Override
024    public boolean equals(final Object o) {
025        return (o instanceof Key) && Arrays.equals(keyBytes, ((Key) o).keyBytes);
026    }
027
028    @Override
029    public int hashCode() {
030        return Arrays.hashCode(keyBytes);
031    }
032
033    @Override
034    public String toString() {
035        return Base64.getEncoder().encodeToString(this.keyBytes);
036    }
037    /*
038    As this object is used as a config object 
039    we add the factory method so empty instances 
040    can be created 
041    */
042    private static Key create() {
043        return new Key(new byte[0]);
044    }
045}
