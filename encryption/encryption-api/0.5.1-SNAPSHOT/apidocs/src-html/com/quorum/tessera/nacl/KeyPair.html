001package com.quorum.tessera.nacl;
002
003import java.util.Objects;
004
005/**
006 * Container object for a public/private key pair
007 *
008 * The public and private key should be related to each other, not just an
009 * arbitrary pairing of two keys
010 */
011public class KeyPair {
012
013    private final Key publicKey;
014
015    private final Key privateKey;
016
017    public KeyPair(final Key publicKey, final Key privateKey) {
018        this.publicKey = Objects.requireNonNull(publicKey);
019        this.privateKey = Objects.requireNonNull(privateKey);
020    }
021
022    public Key getPublicKey() {
023        return publicKey;
024    }
025
026    public Key getPrivateKey() {
027        return privateKey;
028    }
029
030    @Override
031    public boolean equals(final Object o) {
032        return (o instanceof KeyPair) &&
033            Objects.equals(publicKey, ((KeyPair) o).publicKey) &&
034            Objects.equals(privateKey, ((KeyPair) o).privateKey);
035    }
036
037    @Override
038    public int hashCode() {
039        return Objects.hash(getPublicKey(), getPrivateKey());
040    }
041
042}
