001package com.quorum.tessera.config.adapters;
002
003import com.quorum.tessera.config.KeyData;
004import com.quorum.tessera.config.KeyDataConfig;
005import com.quorum.tessera.config.PrivateKeyData;
006import com.quorum.tessera.config.PrivateKeyType;
007import com.quorum.tessera.config.keys.KeyEncryptor;
008import com.quorum.tessera.config.keys.KeyEncryptorFactory;
009import com.quorum.tessera.config.util.IOCallback;
010import com.quorum.tessera.config.util.JaxbUtil;
011import com.quorum.tessera.nacl.NaclException;
012import org.slf4j.Logger;
013import org.slf4j.LoggerFactory;
014
015import javax.xml.bind.annotation.adapters.XmlAdapter;
016import java.io.ByteArrayInputStream;
017import java.nio.file.Files;
018import java.nio.file.Path;
019import java.util.Optional;
020
021import static java.nio.charset.StandardCharsets.UTF_8;
022
023public class KeyDataAdapter extends XmlAdapter<KeyData, KeyData> {
024
025    private static final Logger LOGGER = LoggerFactory.getLogger(KeyDataAdapter.class);
026
027    @Override
028    public KeyData unmarshal(final KeyData keyData) {
029
030        //case 1, the keys are provided inline
031        if (keyData.hasKeys()) {
032            return keyData;
033        }
034
035        //case 2, the config is provided inline
036        if (keyData.getPublicKeyPath() == null && keyData.getPrivateKeyPath() == null) {
037            return unmarshalInline(keyData);
038        }
039
040        if (keyData.getPublicKeyPath() == null || keyData.getPrivateKeyPath() == null
041            || Files.notExists(keyData.getPublicKeyPath()) || Files.notExists(keyData.getPrivateKeyPath())) {
042            System.err.println("When providing key paths, must give both as paths, not just one, and both files must exist");
043            throw new IllegalArgumentException("When providing key paths, must give both public and private, and both files must exist");
044        }
045
046        //case 3, the keys are provided inside a file
047        return unmarshalFile(
048                keyData.getPublicKeyPath(),
049                keyData.getPrivateKeyPath(),
050                Optional.ofNullable(keyData.getConfig()).map(KeyDataConfig::getPassword).orElse(null)
051        );
052    }
053
054    private KeyData unmarshalFile(final Path publicKeyPath, final Path privateKeyPath, final String password) {
055        final byte[] publicKey = IOCallback.execute(() -> Files.readAllBytes(publicKeyPath));
056        final String publicKeyString = new String(publicKey, UTF_8);
057
058        final byte[] privateKey = IOCallback.execute(() -> Files.readAllBytes(privateKeyPath));
059        final String privateKeyString = new String(privateKey, UTF_8);
060
061        final KeyDataConfig unmarshal
062                = JaxbUtil.unmarshal(new ByteArrayInputStream(privateKeyString.getBytes(UTF_8)), KeyDataConfig.class);
063
064        return this.unmarshalInline(
065                new KeyData(
066                        new KeyDataConfig(
067                                new PrivateKeyData(
068                                        unmarshal.getValue(),
069                                        unmarshal.getSnonce(),
070                                        unmarshal.getAsalt(),
071                                        unmarshal.getSbox(),
072                                        unmarshal.getArgonOptions(),
073                                        password
074                                ),
075                                unmarshal.getType()
076                        ),
077                        null,
078                        publicKeyString,
079                        privateKeyPath,
080                        publicKeyPath
081                )
082        );
083
084    }
085
086    private KeyData unmarshalInline(final KeyData keyData) {
087        if (keyData.getConfig().getType() == PrivateKeyType.UNLOCKED) {
088            return new KeyData(keyData.getConfig(), keyData.getConfig().getValue(), keyData.getPublicKey(), null, null);
089        }
090
091        if (keyData.getConfig().getPassword() == null) {
092
093            System.err.println("A locked key was provided without a password. ");
094            System.err.println("Please ensure the same number of passwords are provided as there are keys ");
095            System.err.print("and remember to include empty passwords for unlocked keys");
096            System.err.println();
097            throw new IllegalArgumentException("Password missing");
098        }
099
100        final KeyEncryptor kg = KeyEncryptorFactory.create();
101        final PrivateKeyData encryptedKey = keyData.getConfig().getPrivateKeyData();
102
103        try {
104            //need to decrypt
105            return new KeyData(
106                    keyData.getConfig(),
107                    kg.decryptPrivateKey(encryptedKey).toString(),
108                    keyData.getPublicKey(),
109                    keyData.getPrivateKeyPath(),
110                    keyData.getPublicKeyPath()
111            );
112        } catch (final NaclException ex) {
113            System.err.println("Could not decrypt the private key with the provided password, please double check the passwords provided");
114            throw new IllegalArgumentException();
115        }
116
117    }
118
119    @Override
120    public KeyData marshal(final KeyData keyData) {
121
122        if(keyData.getConfig() == null) {
123            return keyData;
124        }
125
126        if (keyData.getConfig().getType() != PrivateKeyType.UNLOCKED) {
127            return new KeyData(
128                    new KeyDataConfig(
129                            new PrivateKeyData(
130                                    keyData.getConfig().getPrivateKeyData().getValue(),
131                                    keyData.getConfig().getPrivateKeyData().getSnonce(),
132                                    keyData.getConfig().getPrivateKeyData().getAsalt(),
133                                    keyData.getConfig().getPrivateKeyData().getSbox(),
134                                    keyData.getConfig().getPrivateKeyData().getArgonOptions(),
135                                    null
136                            ),
137                            keyData.getConfig().getType()
138                    ),
139                    null,
140                    keyData.getPublicKey(),
141                    null,
142                    null
143            );
144        }
145
146        return keyData;
147    }
148
149}
