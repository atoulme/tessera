001package com.quorum.tessera.config;
002
003import javax.validation.constraints.NotNull;
004import javax.xml.bind.annotation.XmlAccessType;
005import javax.xml.bind.annotation.XmlAccessorType;
006import javax.xml.bind.annotation.XmlElement;
007import javax.xml.bind.annotation.XmlType;
008
009@XmlAccessorType(XmlAccessType.FIELD)
010@XmlType(factoryMethod = "create")
011public class InfluxConfig extends ConfigItem {
012
013    @NotNull
014    @XmlElement(required = true)
015    private final String hostName;
016
017    @NotNull
018    @XmlElement(required = true)
019    private final Integer port;
020
021    @NotNull
022    @XmlElement(required = true)
023    private final Long pushIntervalInSecs;
024
025    @NotNull
026    @XmlElement(required = true)
027    private final String dbName;
028
029    public InfluxConfig(String hostName, Integer port, Long pushIntervalInSecs, String dbName) {
030        this.hostName = hostName;
031        this.port = port;
032        this.dbName = dbName;
033        this.pushIntervalInSecs = pushIntervalInSecs;
034    }
035
036    private static InfluxConfig create() {
037        return new InfluxConfig(null, null, null, null);
038    }
039
040    public String getHostName() {
041        return hostName;
042    }
043
044    public Integer getPort() {
045        return port;
046    }
047
048    public Long getPushIntervalInSecs() {
049        return pushIntervalInSecs;
050    }
051
052    public String getDbName() {
053        return dbName;
054    }
055}
