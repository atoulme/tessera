001package com.quorum.tessera.config.keys;
002
003import com.quorum.tessera.argon2.Argon2;
004import com.quorum.tessera.argon2.ArgonResult;
005import com.quorum.tessera.config.ArgonOptions;
006import com.quorum.tessera.config.PrivateKeyData;
007import com.quorum.tessera.nacl.Key;
008import com.quorum.tessera.nacl.NaclFacade;
009import com.quorum.tessera.nacl.Nonce;
010import org.slf4j.Logger;
011import org.slf4j.LoggerFactory;
012
013import java.security.SecureRandom;
014import java.util.Arrays;
015import java.util.Base64;
016import java.util.Objects;
017
018/**
019 * An implementation of {@link KeyEncryptor} that uses Argon2
020 * <p>
021 * The password is hashed using the generated/provided salt to generate a 32
022 * byte hash This hash is then used as the symmetric key to encrypt the private
023 * key
024 */
025public class KeyEncryptorImpl implements KeyEncryptor {
026
027    private static final Logger LOGGER = LoggerFactory.getLogger(KeyEncryptorImpl.class);
028
029    private final Argon2 argon2;
030
031    private final NaclFacade nacl;
032
033    private final Base64.Decoder decoder = Base64.getDecoder();
034
035    private final Base64.Encoder encoder = Base64.getEncoder();
036
037    private final SecureRandom secureRandom = new SecureRandom();
038
039    public KeyEncryptorImpl(final Argon2 argon2, final NaclFacade nacl) {
040        this.argon2 = Objects.requireNonNull(argon2);
041        this.nacl = Objects.requireNonNull(nacl);
042    }
043
044    @Override
045    public PrivateKeyData encryptPrivateKey(final Key privateKey, final String password, final ArgonOptions argonOptions) {
046
047        LOGGER.info("Encrypting a private key");
048
049        LOGGER.debug("Encrypting private key {} using password {}", privateKey, password);
050
051        final byte[] salt = new byte[KeyEncryptor.SALTLENGTH];
052        this.secureRandom.nextBytes(salt);
053
054        LOGGER.debug("Generated the random salt {}", Arrays.toString(salt));
055
056        final ArgonResult argonResult;
057
058        if(argonOptions == null) {
059            argonResult = this.argon2.hash(password, salt);
060        } else {
061            argonResult = this.argon2.hash(
062                new com.quorum.tessera.argon2.ArgonOptions(
063                    argonOptions.getAlgorithm(),
064                    argonOptions.getIterations(),
065                    argonOptions.getMemory(),
066                    argonOptions.getParallelism()
067                ),
068                password,
069                salt
070            );
071        }
072
073        final Nonce nonce = this.nacl.randomNonce();
074        LOGGER.debug("Generated the random nonce {}", nonce);
075
076        final byte[] encryptedKey = this.nacl.sealAfterPrecomputation(
077            privateKey.getKeyBytes(), nonce, new Key(argonResult.getHash())
078        );
079
080        LOGGER.info("Private key encrypted");
081
082        final String snonce = this.encoder.encodeToString(nonce.getNonceBytes());
083        final String asalt = this.encoder.encodeToString(salt);
084        final String sbox = this.encoder.encodeToString(encryptedKey);
085
086        return new PrivateKeyData(
087            null,
088            snonce,
089            asalt,
090            sbox,
091            new ArgonOptions(
092                argonResult.getOptions().getAlgorithm(),
093                argonResult.getOptions().getIterations(),
094                argonResult.getOptions().getMemory(),
095                argonResult.getOptions().getParallelism()
096            ),
097            null
098        );
099    }
100
101    @Override
102    public Key decryptPrivateKey(final PrivateKeyData privateKey) {
103
104        LOGGER.info("Decrypting private key");
105        LOGGER.debug("Decrypting private key {} using password {}", privateKey.getValue(), privateKey.getPassword());
106
107        final byte[] salt = this.decoder.decode(privateKey.getAsalt());
108
109        final ArgonResult argonResult = this.argon2.hash(
110            new com.quorum.tessera.argon2.ArgonOptions(
111                privateKey.getArgonOptions().getAlgorithm(),
112                privateKey.getArgonOptions().getIterations(),
113                privateKey.getArgonOptions().getMemory(),
114                privateKey.getArgonOptions().getParallelism()
115            ),
116            privateKey.getPassword(),
117            salt
118        );
119
120        final byte[] originalKey = this.nacl.openAfterPrecomputation(
121            this.decoder.decode(privateKey.getSbox()),
122            new Nonce(this.decoder.decode(privateKey.getSnonce())),
123            new Key(argonResult.getHash())
124        );
125
126        LOGGER.info("Decrypting private key");
127        LOGGER.debug("Decrypted private key {}", new Key(originalKey));
128
129        return new Key(originalKey);
130    }
131
132}
