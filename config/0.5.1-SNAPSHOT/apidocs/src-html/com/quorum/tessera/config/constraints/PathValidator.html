001package com.quorum.tessera.config.constraints;
002
003import com.quorum.tessera.config.util.FilesDelegate;
004import java.io.UncheckedIOException;
005import javax.validation.ConstraintValidator;
006import javax.validation.ConstraintValidatorContext;
007import java.nio.file.Path;
008import java.util.Objects;
009import org.slf4j.Logger;
010import org.slf4j.LoggerFactory;
011
012public class PathValidator implements ConstraintValidator<ValidPath, Path> {
013
014    private static final Logger LOGGER = LoggerFactory.getLogger(PathValidator.class);
015    
016    private ValidPath config;
017    
018    private  FilesDelegate filesDelegate = FilesDelegate.create();
019    
020    @Override
021    public void initialize(ValidPath a) {
022        this.config = a;
023    }
024
025    @Override
026    public boolean isValid(Path t, ConstraintValidatorContext constraintContext) {
027        //Not null deals with this
028        if (Objects.isNull(t)) {
029            return true;
030        }
031
032        if (config.checkCanCreate() && filesDelegate.notExists(t)) {
033            try {
034                filesDelegate.createFile(t);
035            } catch (UncheckedIOException ex) {
036                LOGGER.debug(null, ex);
037                constraintContext.disableDefaultConstraintViolation();
038                constraintContext.buildConstraintViolationWithTemplate("Unable to create file "+ t)
039                        .addConstraintViolation();
040                return false;
041            } finally {
042                try {
043                    filesDelegate.deleteIfExists(t);
044                } catch (UncheckedIOException ex) {
045                    LOGGER.trace(null, ex);
046                    //Not much we can do
047                }
048            }
049
050        }
051
052        return !config.checkExists() || !filesDelegate.notExists(t);
053    }
054
055    /*
056    TEsting only needs factory
057    */
058    protected void setFilesDelegate(FilesDelegate filesDelegate) {
059        this.filesDelegate = filesDelegate;
060    }
061    
062}
