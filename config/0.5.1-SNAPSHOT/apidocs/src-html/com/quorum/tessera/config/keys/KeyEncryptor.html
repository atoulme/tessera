001package com.quorum.tessera.config.keys;
002
003import com.quorum.tessera.config.ArgonOptions;
004import com.quorum.tessera.config.PrivateKeyData;
005import com.quorum.tessera.nacl.Key;
006
007/**
008 * Provides key encryption and decryption for private keys
009 * into the representation expected by a private key file
010 */
011public interface KeyEncryptor {
012
013    int SALTLENGTH = 16;
014
015    /**
016     * Encrypts a private key using the given password and returns a JSON object that can
017     * be interpreted when decrypting
018     *
019     * @param privateKey the key to encrypt
020     * @param password   the password to encrypt the key with
021     * @return the configuration that can be used to decrypt the private key
022     */
023    PrivateKeyData encryptPrivateKey(Key privateKey, String password, ArgonOptions argonOptions);
024
025    /**
026     * Decrypts a private key using the password and information provided by the given
027     * JSON object. What information the object contains is up to the implementor.
028     *
029     * @param privateKeyConfig the configuration used to decrypt the private keyt
030     * @return the decrypted private key
031     */
032    Key decryptPrivateKey(PrivateKeyData privateKeyConfig);
033
034}
