001package com.quorum.tessera.config;
002
003import java.util.Iterator;
004import java.util.Optional;
005import java.util.ServiceLoader;
006
007public interface ServiceLoaderUtil {
008    
009    static <T> Optional<T> load(Class<T> type) {
010        Iterator<T> it =  ServiceLoader.load(type).iterator();
011        if(it.hasNext()) {
012            return Optional.of(it.next());
013        }
014        return Optional.empty();
015    }
016    
017}
