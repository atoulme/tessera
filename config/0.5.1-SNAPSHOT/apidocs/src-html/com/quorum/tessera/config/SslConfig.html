090    public SslConfig(

107            Path serverTlsKeyPath,
108            Path serverTlsCertificatePath,
109            Path clientTlsKeyPath,
110            Path clientTlsCertificatePath
111    ) {
112
113        this.tls = tls;
114        this.generateKeyStoreIfNotExisted = generateKeyStoreIfNotExisted;
115        this.serverKeyStore = serverKeyStore;
116        this.serverTrustCertificates = serverTrustCertificates;
117        this.serverKeyStorePassword = serverKeyStorePassword;

130        this.serverTlsCertificatePath = serverTlsCertificatePath;

132        this.clientTlsCertificatePath = clientTlsCertificatePath;
133
134    }
