001package com.quorum.tessera.config;
002
003import java.util.Optional;
004import javax.validation.constraints.NotNull;
005import javax.validation.constraints.Pattern;
006import javax.xml.bind.annotation.XmlAccessType;
007import javax.xml.bind.annotation.XmlAccessorType;
008import javax.xml.bind.annotation.XmlAttribute;
009import javax.xml.bind.annotation.XmlType;
010
011@XmlType(factoryMethod = "create")
012@XmlAccessorType(XmlAccessType.FIELD)
013public class ArgonOptions extends ConfigItem {
014
015    @Pattern(regexp = "^(id|i|d)$")
016    @XmlAttribute(name = "variant")
017    private final String algorithm;
018
019    @NotNull
020    @XmlAttribute
021    private final Integer iterations;
022
023    @NotNull
024    @XmlAttribute
025    private final Integer memory;
026
027    @NotNull
028    @XmlAttribute
029    private Integer parallelism;
030
031    private static ArgonOptions create() {
032        return new ArgonOptions();
033    }
034
035    public ArgonOptions(String algorithm, Integer iterations, Integer memory, Integer parallelism) {
036        this.algorithm = Optional.ofNullable(algorithm).orElse("id");
037        this.iterations = iterations;
038        this.memory = memory;
039        this.parallelism = parallelism;
040    }
041
042    private ArgonOptions() {
043        this(null, null, null, null);
044    }
045
046    public String getAlgorithm() {
047        return algorithm;
048    }
049
050    public Integer getIterations() {
051        return iterations;
052    }
053
054    public Integer getMemory() {
055        return memory;
056    }
057
058    public Integer getParallelism() {
059        return parallelism;
060    }
061
062}
