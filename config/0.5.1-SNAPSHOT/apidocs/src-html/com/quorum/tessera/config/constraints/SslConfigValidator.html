001package com.quorum.tessera.config.constraints;
002
003
004import com.quorum.tessera.config.SslAuthenticationMode;
005import com.quorum.tessera.config.SslConfig;
006import com.quorum.tessera.config.SslTrustMode;
007
008import javax.validation.ConstraintValidator;
009import javax.validation.ConstraintValidatorContext;
010import java.nio.file.Files;
011import java.util.Objects;
012
013public class SslConfigValidator implements ConstraintValidator<ValidSsl,SslConfig> {
014
015    private ValidSsl validSsl;
016
017    @Override
018    public void initialize(ValidSsl validSsl) {
019        this.validSsl = validSsl;
020    }
021
022    @Override
023    public boolean isValid(SslConfig sslConfig, ConstraintValidatorContext context) {
024
025        context.disableDefaultConstraintViolation();
026
027        if (validSsl.checkSslValid()){
028
029            if (Objects.isNull(sslConfig)) {
030                return true;
031            }
032
033            if (sslConfig.getTls() == SslAuthenticationMode.STRICT) {
034
035                if (!sslConfig.isGenerateKeyStoreIfNotExisted()) {
036
037                    if (!isServerKeyStoreConfigValid(sslConfig, context) ||
038                        !isClientKeyStoreConfigValid(sslConfig, context)) {
039                        return false;
040                    }
041                }
042
043                if (!isTrustModeConfigValid(sslConfig, context)) {
044                    return false;
045                }
046
047                if (!isServerConfigValidForWhiteListMode(sslConfig, context)) {
048                    return false;
049                }
050
051                if (!isServerConfigValidForCAMode(sslConfig, context)) {
052                    return false;
053                }
054
055                if (!isClientConfigValidForWhiteListMode(sslConfig, context)) {
056                    return false;
057                }
058
059                if (!isClientConfigValidForCAMode(sslConfig, context)) {
060                    return false;
061                }
062            }
063        }
064        return true;
065    }
066
067    private boolean isServerKeyStoreConfigValid(SslConfig sslConfig, ConstraintValidatorContext context) {
068        if (Objects.isNull(sslConfig.getServerKeyStore()) ||
069            Objects.isNull(sslConfig.getServerKeyStorePassword()) ||
070            Files.notExists(sslConfig.getServerKeyStore())) {
071            if (Objects.isNull(sslConfig.getServerTlsKeyPath()) ||
072                Objects.isNull(sslConfig.getServerTlsCertificatePath()) ||
073                Files.notExists(sslConfig.getServerTlsKeyPath()) ||
074                Files.notExists(sslConfig.getServerTlsCertificatePath())) {
075                setMessage("Server keystore configuration not valid. " +
076                    "Please ensure keystore file exists or keystore password not null, " +
077                    "otherwise please set keystore generation flag to true to have keystore created", context);
078                return false;
079            }
080        }
081        return true;
082    }
083
084    private boolean isClientKeyStoreConfigValid(SslConfig sslConfig, ConstraintValidatorContext context) {
085        if (Objects.isNull(sslConfig.getClientKeyStore()) ||
086            Objects.isNull(sslConfig.getClientKeyStorePassword()) ||
087            Files.notExists(sslConfig.getClientKeyStore())) {
088            if (Objects.isNull(sslConfig.getClientTlsKeyPath()) ||
089                Objects.isNull(sslConfig.getClientTlsCertificatePath()) ||
090                Files.notExists(sslConfig.getClientTlsKeyPath()) ||
091                Files.notExists(sslConfig.getClientTlsCertificatePath())) {
092                setMessage("Client keystore configuration not valid. " +
093                    "Please ensure keystore file exists or keystore password not null, " +
094                    "otherwise please set keystore generation flag to true to have keystore created", context);
095                return false;
096            }
097        }
098        return true;
099    }
100
101    private boolean isTrustModeConfigValid(SslConfig sslConfig, ConstraintValidatorContext context) {
102        if (Objects.isNull(sslConfig.getServerTrustMode()) || Objects.isNull(sslConfig.getClientTrustMode())) {
103            setMessage("Trust mode does not have valid value. Please check server/client trust mode config", context);
104            return false;
105        }
106        return true;
107    }
108
109    private boolean isServerConfigValidForWhiteListMode(SslConfig sslConfig, ConstraintValidatorContext context) {
110        if (sslConfig.getServerTrustMode() == SslTrustMode.WHITELIST) {
111            if (Objects.isNull(sslConfig.getKnownClientsFile()) ||
112                Files.notExists(sslConfig.getKnownClientsFile())) {
113                setMessage("Known clients file not found. If server trust mode is WHITELIST, known clients file must be provided", context);
114                return false;
115            }
116        }
117        return true;
118    }
119
120    private boolean isServerConfigValidForCAMode(SslConfig sslConfig, ConstraintValidatorContext context) {
121        if (sslConfig.getServerTrustMode() == SslTrustMode.CA) {
122            if (Objects.isNull(sslConfig.getServerTrustStore()) ||
123                Objects.isNull(sslConfig.getServerTrustStorePassword()) ||
124                Files.notExists(sslConfig.getServerTrustStore())) {
125                if (Objects.isNull(sslConfig.getServerTrustCertificates())) {
126                    setMessage("Trust store config not valid. If server trust mode is CA, trust store must exist and not be null", context);
127                    return false;
128                }
129            }
130        }
131        return true;
132    }
133
134    private boolean isClientConfigValidForWhiteListMode(SslConfig sslConfig, ConstraintValidatorContext context) {
135        if (sslConfig.getClientTrustMode() == SslTrustMode.WHITELIST) {
136            if (Objects.isNull(sslConfig.getKnownServersFile()) ||
137                Files.notExists(sslConfig.getKnownServersFile())) {
138                setMessage("Known servers file not found. If client trust mode is WHITELIST, known servers file must be provided", context);
139                return false;
140            }
141        }
142        return true;
143    }
144
145    private boolean isClientConfigValidForCAMode(SslConfig sslConfig, ConstraintValidatorContext context) {
146        if (sslConfig.getClientTrustMode() == SslTrustMode.CA) {
147            if (Objects.isNull(sslConfig.getClientTrustStore()) ||
148                Objects.isNull(sslConfig.getClientTrustStorePassword()) ||
149                Files.notExists(sslConfig.getClientTrustStore())) {
150                if (Objects.isNull(sslConfig.getClientTrustCertificates())) {
151                    setMessage("Trust store config not valid. If client trust mode is CA, trust store must exist and not be null", context);
152                    return false;
153                }
154            }
155        }
156        return true;
157    }
158
159
160
161    private void setMessage(final String message, ConstraintValidatorContext context) {
162        context
163            .buildConstraintViolationWithTemplate(message)
164            .addConstraintViolation();
165    }
166}
