001package com.quorum.tessera.config;
002
003import com.quorum.tessera.config.util.JaxbUtil;
004import com.quorum.tessera.config.keys.KeyGenerator;
005import com.quorum.tessera.config.keys.KeyGeneratorFactory;
006
007import java.io.InputStream;
008import java.util.List;
009import java.util.Objects;
010import java.util.stream.Collectors;
011import java.util.stream.Stream;
012
013public class JaxbConfigFactory implements ConfigFactory {
014
015
016    private final KeyGenerator generator = KeyGeneratorFactory.newFactory().create();
017    
018    @Override
019    public Config create(final InputStream configData, final ArgonOptions keygenConfig, final String... filenames) {
020
021        final List<KeyData> newKeys = Stream
022            .of(filenames)
023            .map(name -> generator.generate(name, keygenConfig))
024            .map(kd -> new KeyData(
025                    new KeyDataConfig(
026                        new PrivateKeyData(
027                            kd.getConfig().getValue(),
028                            kd.getConfig().getSnonce(),
029                            kd.getConfig().getAsalt(),
030                            kd.getConfig().getSbox(),
031                            kd.getConfig().getArgonOptions(),
032                            null
033                        ),
034                        kd.getConfig().getType()
035                    ),
036                    kd.getPrivateKey(),
037                    kd.getPublicKey(),
038                    kd.getPrivateKeyPath(),
039                    kd.getPublicKeyPath()
040                )
041            )
042            .collect(Collectors.toList());
043
044
045
046        final Config config = JaxbUtil.unmarshal(configData, Config.class);
047
048        if (Objects.nonNull(config.getKeys())) {
049            config.getKeys().getKeyData().addAll(newKeys);
050        }
051
052        return config;
053    }
054
055}
