001package com.quorum.tessera.config.util;
002
003import com.quorum.tessera.config.ServiceLoaderUtil;
004import java.nio.file.Files;
005import java.nio.file.LinkOption;
006import java.nio.file.Path;
007import java.nio.file.attribute.FileAttribute;
008
009public interface FilesDelegate {
010
011    /**
012     * @see java.nio.file.Files#notExists(java.nio.file.Path,
013     * java.nio.file.LinkOption...)
014     */
015    default boolean notExists(Path path, LinkOption... options) {
016        return Files.notExists(path, options);
017    }
018
019    /**
020     * @see java.nio.file.Files#deleteIfExists(java.nio.file.Path)
021     */
022    default boolean deleteIfExists(Path path) {
023        return IOCallback.execute(() -> Files.deleteIfExists(path));
024    }
025
026    /**
027     * @see java.nio.file.Files#createFile(java.nio.file.Path, java.nio.file.attribute.FileAttribute...) 
028     */
029    default Path createFile(Path path, FileAttribute... attributes) {
030
031        return IOCallback.execute(() -> Files.createFile(path, attributes));
032
033    }
034
035    static FilesDelegate create() {
036        return ServiceLoaderUtil.load(FilesDelegate.class).orElse(new FilesDelegate() {
037        });
038    }
039
040}
