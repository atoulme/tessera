001package com.quorum.tessera.config.util;
002
003import java.io.Console;
004import java.io.InputStream;
005import java.util.Scanner;
006
007/**
008 * Allows a password to be read from the System console if it is available
009 * otherwise reads from the provided input stream
010 */
011public class PasswordReader {
012
013    private final Console console;
014
015    private final Scanner systemIn;
016
017    public PasswordReader(final Console console, final InputStream systemIn) {
018        this.console = console;
019        this.systemIn = new Scanner(systemIn);
020    }
021
022    /**
023     * Read a password from either system console or the given stream
024     * @return the read password, which may be empty if no password is given
025     */
026    public String readPassword() {
027        if(this.console == null) {
028            return this.systemIn.nextLine();
029        }
030
031        final char[] consolePassword = this.console.readPassword();
032        return new String(consolePassword);
033    }
034
035}
