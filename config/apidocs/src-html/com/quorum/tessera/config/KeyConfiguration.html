001package com.quorum.tessera.config;
002
003import com.quorum.tessera.config.adapters.PathAdapter;
004
005import javax.validation.constraints.NotNull;
006import javax.xml.bind.annotation.XmlAccessType;
007import javax.xml.bind.annotation.XmlAccessorType;
008import javax.xml.bind.annotation.XmlElement;
009import javax.xml.bind.annotation.XmlType;
010import javax.xml.bind.annotation.adapters.XmlJavaTypeAdapter;
011import java.nio.file.Path;
012import java.util.List;
013
014@XmlAccessorType(XmlAccessType.FIELD)
015@XmlType(factoryMethod = "create")
016public class KeyConfiguration extends ConfigItem {
017
018    @XmlElement(type = String.class)
019    @XmlJavaTypeAdapter(PathAdapter.class)
020    private final Path passwordFile;
021
022    private final List<String> passwords;
023
024    @NotNull
025    private final List<KeyData> keyData;
026
027    public KeyConfiguration(final Path passwordFile, final List<String> passwords, final List<KeyData> keyData) {
028        this.passwordFile = passwordFile;
029        this.passwords = passwords;
030        this.keyData = keyData;
031    }
032
033    private static KeyConfiguration create() {
034        return new KeyConfiguration(null, null, null);
035    }
036
037    public Path getPasswordFile() {
038        return this.passwordFile;
039    }
040
041    public List<String> getPasswords() {
042        return this.passwords;
043    }
044
045    public List<KeyData> getKeyData() {
046        return this.keyData;
047    }
048
049}
