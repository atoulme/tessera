001package com.quorum.tessera.config.adapters;
002
003import com.quorum.tessera.config.PrivateKeyType;
004
005import javax.xml.bind.annotation.adapters.XmlAdapter;
006import java.util.HashMap;
007import java.util.Map;
008
009public class PrivateKeyTypeAdapter extends XmlAdapter<String, PrivateKeyType> {
010
011    private static final Map<String, PrivateKeyType> MAPPING = new HashMap<String, PrivateKeyType>() {
012        {
013            put("unlocked", PrivateKeyType.UNLOCKED);
014            put("argon2sbox", PrivateKeyType.LOCKED);
015        }
016    };
017
018    @Override
019    public PrivateKeyType unmarshal(final String v) {
020        return MAPPING.get(v);
021    }
022
023    @Override
024    public String marshal(final PrivateKeyType v) {
025        return MAPPING
026            .entrySet()
027            .stream()
028            .filter(e -> e.getValue() == v)
029            .map(Map.Entry::getKey)
030            .findAny()
031            .orElse(null);
032    }
033
034}
