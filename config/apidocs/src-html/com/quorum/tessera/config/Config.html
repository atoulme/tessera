001package com.quorum.tessera.config;
002
003import com.quorum.tessera.config.adapters.KeyAdapter;
004import com.quorum.tessera.config.adapters.KeyConfigurationAdapter;
005import com.quorum.tessera.config.adapters.PathAdapter;
006import com.quorum.tessera.nacl.Key;
007
008import javax.validation.Valid;
009import javax.validation.constraints.NotNull;
010import javax.validation.constraints.Size;
011import javax.xml.bind.annotation.*;
012import javax.xml.bind.annotation.adapters.XmlJavaTypeAdapter;
013import java.nio.file.Path;
014import java.util.List;
015
016@XmlRootElement
017@XmlAccessorType(XmlAccessType.FIELD)
018@XmlType(factoryMethod = "create")
019public class Config extends ConfigItem {
020
021    @NotNull
022    @Valid
023    @XmlElement(name = "jdbc", required = true)
024    private final JdbcConfig jdbcConfig;
025
026    @NotNull
027    @Valid
028    @XmlElement(name = "server", required = true)
029    private final ServerConfig serverConfig;
030
031    @NotNull
032    @Size(min = 1)
033    @Valid
034    @XmlElement(name = "peer", required = true)
035    private final List<Peer> peers;
036
037    @Valid
038    @NotNull
039    @XmlElement(required = true)
040    @XmlJavaTypeAdapter(KeyConfigurationAdapter.class)
041    private final KeyConfiguration keys;
042
043    @NotNull
044    @XmlElement(name = "alwaysSendTo", required = true)
045    @XmlJavaTypeAdapter(KeyAdapter.class)
046    private final List<Key> alwaysSendTo;
047
048    @NotNull
049    @XmlElement(required = true, type = String.class)
050    @XmlJavaTypeAdapter(PathAdapter.class)
051    private final Path unixSocketFile;
052
053    @XmlAttribute
054    private final boolean useWhiteList;
055
056    public Config(final JdbcConfig jdbcConfig,
057                  final ServerConfig serverConfig,
058                  final List<Peer> peers,
059                  final KeyConfiguration keyConfiguration,
060                  final List<Key> alwaysSendTo,
061                  final Path unixSocketFile,
062                  final boolean useWhiteList) {
063        this.jdbcConfig = jdbcConfig;
064        this.serverConfig = serverConfig;
065        this.peers = peers;
066        this.keys = keyConfiguration;
067        this.alwaysSendTo = alwaysSendTo;
068        this.unixSocketFile = unixSocketFile;
069        this.useWhiteList = useWhiteList;
070    }
071
072    private static Config create() {
073        return new Config();
074    }
075
076    private Config() {
077        this(null, null, null, null, null, null, false);
078    }
079
080    public JdbcConfig getJdbcConfig() {
081        return this.jdbcConfig;
082    }
083
084    public ServerConfig getServerConfig() {
085        return this.serverConfig;
086    }
087
088    public Path getUnixSocketFile() {
089        return this.unixSocketFile;
090    }
091
092    public List<Peer> getPeers() {
093        return this.peers;
094    }
095
096    public KeyConfiguration getKeys() {
097        return this.keys;
098    }
099
100    public List<Key> getAlwaysSendTo() {
101        return this.alwaysSendTo;
102    }
103
104    public boolean isUseWhiteList() {
105        return this.useWhiteList;
106    }
107
108}
