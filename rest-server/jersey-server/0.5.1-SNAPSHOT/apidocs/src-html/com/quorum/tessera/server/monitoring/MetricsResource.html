001package com.quorum.tessera.server.monitoring;
002
003import javax.management.*;
004import javax.ws.rs.GET;
005import javax.ws.rs.Path;
006import javax.ws.rs.Produces;
007import javax.ws.rs.core.Response;
008import java.lang.management.ManagementFactory;
009import java.util.List;
010
011import static javax.ws.rs.core.MediaType.TEXT_PLAIN;
012
013@Path("/metrics")
014public class MetricsResource {
015
016    private final MBeanServer mbs;
017
018    public MetricsResource() {
019        mbs = ManagementFactory.getPlatformMBeanServer();
020    }
021
022    @GET
023    @Produces("text/plain")
024    public Response getMetrics() {
025        MetricsEnquirer metricsEnquirer = new MetricsEnquirer(mbs);
026        List<MBeanMetric> metrics = metricsEnquirer.getMBeanMetrics();
027
028        PrometheusProtocolFormatter formatter = new PrometheusProtocolFormatter();
029        String formattedMetrics = formatter.format(metrics);
030
031        return Response.status(Response.Status.OK)
032            .header("Content-Type", TEXT_PLAIN)
033            .entity(formattedMetrics)
034            .build();
035    }
036}
