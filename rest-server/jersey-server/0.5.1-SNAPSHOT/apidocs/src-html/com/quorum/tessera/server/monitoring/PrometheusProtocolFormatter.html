001package com.quorum.tessera.server.monitoring;
002
003import java.util.List;
004
005public class PrometheusProtocolFormatter {
006
007    public String format(List<MBeanMetric> metrics) {
008        StringBuilder formattedMetrics = new StringBuilder();
009
010        for(MBeanMetric metric : metrics) {
011            MBeanResourceMetric resourceMetric = (MBeanResourceMetric) metric;
012
013            formattedMetrics.append("tessera_")
014                            .append(sanitize(resourceMetric.getResourceMethod()))
015                            .append("_")
016                            .append(sanitize(resourceMetric.getName()))
017                            .append(" ")
018                            .append(resourceMetric.getValue())
019                            .append("\n");
020        }
021
022        return formattedMetrics.toString().trim();
023    }
024
025    private String sanitize(String input) {
026        return input.replaceAll("(#.*)|(_total)|\\(\\)|\\)|\\[\\]|\\]|;", "")
027                    .replaceAll("->|\\(|\\[", "_");
028    }
029
030
031}
