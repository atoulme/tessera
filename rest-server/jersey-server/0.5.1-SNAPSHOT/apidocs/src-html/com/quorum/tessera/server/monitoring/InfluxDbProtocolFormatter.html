001package com.quorum.tessera.server.monitoring;
002
003import java.net.URI;
004import java.util.List;
005
006public class InfluxDbProtocolFormatter {
007
008    public String format(List<MBeanMetric> metrics, URI uri) {
009        StringBuilder formattedMetrics = new StringBuilder();
010
011        for(MBeanMetric metric : metrics) {
012            MBeanResourceMetric resourceMetric = (MBeanResourceMetric) metric;
013
014            formattedMetrics.append("tessera_")
015                            .append(sanitize(resourceMetric.getResourceMethod()))
016                            .append(",")
017                            .append("instance=")
018                            .append(uri.getHost())
019                            .append(":")
020                            .append(uri.getPort())
021                            .append(" ")
022                            .append(sanitize(resourceMetric.getName()))
023                            .append("=")
024                            .append(resourceMetric.getValue())
025                            .append("\n");
026        }
027
028        return formattedMetrics.toString().trim();
029    }
030
031    private String sanitize(String input) {
032        return input.replaceAll("(#.*)|(_total)|\\(\\)|\\)|\\[\\]|\\]|;", "")
033                    .replaceAll("->|\\(|\\[", "_");
034    }
035}
