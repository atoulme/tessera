001package com.quorum.tessera.server;
002
003import com.quorum.tessera.config.InfluxConfig;
004import com.quorum.tessera.config.ServerConfig;
005import com.quorum.tessera.server.monitoring.InfluxDbClient;
006import com.quorum.tessera.server.monitoring.InfluxDbPublisher;
007import com.quorum.tessera.server.monitoring.MetricsResource;
008import com.quorum.tessera.ssl.context.SSLContextFactory;
009import com.quorum.tessera.ssl.context.ServerSSLContextFactory;
010import org.glassfish.grizzly.http.server.HttpServer;
011import org.glassfish.grizzly.servlet.ServletRegistration;
012import org.glassfish.grizzly.servlet.WebappContext;
013import org.glassfish.grizzly.ssl.SSLEngineConfigurator;
014import org.glassfish.jersey.grizzly2.httpserver.GrizzlyHttpServerFactory;
015import org.glassfish.jersey.server.ResourceConfig;
016import org.glassfish.jersey.servlet.ServletContainer;
017import org.slf4j.Logger;
018import org.slf4j.LoggerFactory;
019import org.slf4j.bridge.SLF4JBridgeHandler;
020
021import javax.net.ssl.SSLContext;
022import javax.ws.rs.core.Application;
023import java.net.URI;
024import java.util.HashMap;
025import java.util.Map;
026import java.util.Objects;
027import java.util.concurrent.ScheduledExecutorService;
028import java.util.concurrent.TimeUnit;
029
030import static java.util.concurrent.Executors.newSingleThreadScheduledExecutor;
031
032/**
033 * Implementation of a RestServer using Jersey and Grizzly.
034 */
035public class JerseyServer implements RestServer {
036
037    private static final Logger LOGGER = LoggerFactory.getLogger(JerseyServer.class);
038
039    private HttpServer server;
040
041    private final URI uri;
042
043    private final Application application;
044
045    private final SSLContext sslContext;
046
047    private final boolean secure;
048
049    private final ScheduledExecutorService executor;
050
051    private final InfluxConfig influxConfig;
052
053    public JerseyServer(final URI uri, final Application application, final ServerConfig serverConfig) {
054        this.uri = Objects.requireNonNull(uri);
055        this.application = Objects.requireNonNull(application);
056        this.secure = serverConfig.isSsl();
057
058        if (this.secure) {
059            final SSLContextFactory sslContextFactory = ServerSSLContextFactory.create();
060            this.sslContext = sslContextFactory.from(uri.toString(), serverConfig.getSslConfig());
061        } else {
062            this.sslContext = null;
063        }
064
065        this.executor = newSingleThreadScheduledExecutor();
066
067        if(serverConfig.getInfluxConfig() != null) {
068            this.influxConfig = serverConfig.getInfluxConfig();
069        } else {
070            this.influxConfig = null;
071        }
072    }
073
074    @Override
075    public void start() throws Exception {
076
077        SLF4JBridgeHandler.removeHandlersForRootLogger();
078        SLF4JBridgeHandler.install();
079
080        //https://jersey.github.io/documentation/latest/appendix-properties.html
081        final Map<String, Object> initParams = new HashMap<>();
082        initParams.put("jersey.config.server.application.name", application.getClass().getSimpleName());
083        initParams.put("jersey.config.server.tracing.type", "ON_DEMAND");
084        initParams.put("jersey.config.server.tracing.threshold", "SUMMARY");
085        initParams.put("jersey.config.logging.verbosity", "PAYLOAD_ANY");
086        initParams.put("jersey.config.beanValidation.enableOutputValidationErrorEntity.server", "true");
087        initParams.put("jersey.config.server.monitoring.statistics.enabled", "true");
088        initParams.put("jersey.config.server.monitoring.enabled", "true");
089        initParams.put("jersey.config.server.monitoring.statistics.mbeans.enabled", "true");
090
091        final ResourceConfig config = ResourceConfig.forApplication(application);
092        config.addProperties(initParams)
093              .register(MetricsResource.class);
094
095        if (this.secure) {
096            this.server = GrizzlyHttpServerFactory.createHttpServer(
097                uri,
098                new ResourceConfig(),
099                true,
100                new SSLEngineConfigurator(sslContext).setClientMode(false).setNeedClientAuth(true),
101                false
102            );
103        } else {
104            this.server = GrizzlyHttpServerFactory.createHttpServer(uri, false);
105        }
106
107        final WebappContext ctx = new WebappContext("WebappContext");
108        final ServletRegistration registration = ctx.addServlet("ServletContainer", new ServletContainer(config));
109        registration.addMapping("/*");
110
111        ctx.deploy(this.server);
112
113        LOGGER.info("Starting {}", uri);
114
115        this.server.start();
116
117        LOGGER.info("Started {}", uri);
118        LOGGER.info("WADL {}/application.wadl", uri);
119
120        if(influxConfig != null) {
121            startInfluxMonitoring();
122        }
123
124    }
125
126    private void startInfluxMonitoring() {
127        InfluxDbClient influxDbClient = new InfluxDbClient(this.uri, influxConfig);
128        Runnable publisher = new InfluxDbPublisher(influxDbClient);
129
130        final Runnable exceptionSafePublisher = () -> {
131            try {
132                publisher.run();
133            } catch (final Throwable ex) {
134                LOGGER.error("Error when executing action {}", publisher.getClass().getSimpleName());
135                LOGGER.error("Error when executing action", ex);
136            }
137        };
138
139        final long delayInSecs = influxConfig.getPushIntervalInSecs();
140        this.executor.scheduleWithFixedDelay(exceptionSafePublisher, delayInSecs, delayInSecs, TimeUnit.SECONDS);
141    }
142
143    @Override
144    public void stop() {
145        LOGGER.info("Stopping Jersey server at {}", uri);
146
147        if(influxConfig != null) {
148            this.executor.shutdown();
149        }
150
151        if (Objects.nonNull(this.server)) {
152            this.server.shutdown();
153        }
154
155        LOGGER.info("Stopped Jersey server at {}", uri);
156    }
157
158}
