001package com.quorum.tessera.server.monitoring;
002
003import org.slf4j.Logger;
004import org.slf4j.LoggerFactory;
005
006public class InfluxDbPublisher implements Runnable {
007
008    private static final Logger LOGGER = LoggerFactory.getLogger(InfluxDbPublisher.class);
009
010    private final InfluxDbClient client;
011
012    public InfluxDbPublisher(InfluxDbClient client) {
013        this.client = client;
014    }
015
016    @Override
017    public void run() {
018        LOGGER.info("InfluxDbPublisher executed...");
019        client.postMetrics();
020    }
021}
