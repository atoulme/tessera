026    public static ConfigBuilder from(Config config) {
027
028        final ConfigBuilder configBuilder = ConfigBuilder.create();
029        configBuilder.unixSocketFile(config.getUnixSocketFile());
030
031        List<String> peers = Stream.of(config)
032            .filter(c -> c.getPeers() != null)
033            .map(Config::getPeers)
034            .flatMap(List::stream)
035            .map(Peer::getUrl)
036            .collect(Collectors.toList());
037
038        configBuilder.jdbcConfig(config.getJdbcConfig())
039                .peers(peers)
040                .serverHostname(config.getServerConfig().getHostName())
041                .serverPort(config.getServerConfig().getPort())
042                .useWhiteList(config.isUseWhiteList());
043
044        final SslConfig sslConfig = config.getServerConfig().getSslConfig();
045
046
047        configBuilder.sslAuthenticationMode(sslConfig.getTls())
048                .sslClientTrustMode(sslConfig.getClientTrustMode())
049                .sslClientKeyStorePath(Objects.toString(sslConfig.getClientKeyStore(), null))
050                .sslClientKeyStorePassword(sslConfig.getClientKeyStorePassword())
051                .sslClientTrustStorePath(Objects.toString(sslConfig.getClientTrustStore(), null))
052                .sslClientTrustStorePassword(sslConfig.getClientTrustStorePassword())
053                .sslClientTlsKeyPath(sslConfig.getClientTlsKeyPath())
054                .sslClientTlsCertificatePath(sslConfig.getClientTlsCertificatePath())
055                .sslClientTrustCertificates(Objects.isNull(sslConfig.getClientTrustCertificates()) ?
056                    EMPTY_LIST :
057                    sslConfig.getClientTrustCertificates()
058                )
059                .sslKnownServersFile(sslConfig.getKnownServersFile())
060                .sslServerTrustMode(sslConfig.getServerTrustMode())
061                .sslServerKeyStorePath(Objects.toString(sslConfig.getServerKeyStore(), null))
062                .sslServerKeyStorePassword(sslConfig.getServerKeyStorePassword())
063                .sslServerTrustStorePath(Objects.toString(sslConfig.getServerTrustStore(), null))
064                .sslServerTrustStorePassword(sslConfig.getServerTrustStorePassword())
065                .sslServerTlsKeyPath(sslConfig.getServerTlsKeyPath())
066                .sslServerTlsCertificatePath(sslConfig.getServerTlsCertificatePath())
067                .sslServerTrustCertificates(Objects.isNull(sslConfig.getServerTrustCertificates()) ?
068                    EMPTY_LIST :
069                    sslConfig.getServerTrustCertificates()
070                )
071                .sslKnownClientsFile(sslConfig.getKnownClientsFile())
072                .sslKnownClientsFile(sslConfig.getKnownClientsFile())
073                .sslKnownServersFile(sslConfig.getKnownServersFile())
074                .sslClientTlsCertificatePath(sslConfig.getClientTlsCertificatePath())
075                .sslServerTlsCertificatePath(sslConfig.getServerTlsCertificatePath())
076                .keyData(config.getKeys())
077                .sslClientTlsKeyPath(sslConfig.getClientTlsKeyPath())
078                .sslServerTlsKeyPath(sslConfig.getServerTlsKeyPath())
079                .alwaysSendToKeys(config.getAlwaysSendTo());
080
081        return configBuilder;
082
083    }

281    static Path toPath(String value) {
282        return Optional.ofNullable(value)
283                .map(Paths::get)
284                .orElse(null);
285    }

287    public Config build() {
288
289        boolean generateKeyStoreIfNotExisted = true;
290
291        SslConfig sslConfig = new SslConfig(
292                sslAuthenticationMode,
293                generateKeyStoreIfNotExisted,
294                toPath(sslServerKeyStorePath),
295                sslServerKeyStorePassword,
296                toPath(sslServerTrustStorePath),
297                sslServerTrustStorePassword,
298                sslServerTrustMode,
299                toPath(sslClientKeyStorePath),
300                sslClientKeyStorePassword,
301                toPath(sslClientTrustStorePath),
302                sslClientTrustStorePassword,
303                sslClientTrustMode,
304                sslKnownClientsFile,
305                sslKnownServersFile,
306                sslServerTrustCertificates.stream()
307                        .filter(Objects::nonNull)
308                        .collect(Collectors.toList()),
309                sslClientTrustCertificates.stream()
310                        .filter(Objects::nonNull)
311                        .collect(Collectors.toList()),
312                sslServerTlsKeyPath,
313                sslServerTlsCertificatePath,
314                sslClientTlsKeyPath,
315                sslClientTlsCertificatePath
316        );
317
318        final ServerConfig serverConfig = new ServerConfig(serverHostname, serverPort, sslConfig, null);
319
320        final List<Peer> peerList = peers
321            .stream()
322                .map(Peer::new)
323                .collect(Collectors.toList());
324
325        final List<Key> forwardingKeys;
326        if(alwaysSendTo != null) {
327            List<String> keyList = new ArrayList<>();
328
329            for(String keyPath : alwaysSendTo) {
330                try {
331                    List<String> keysFromFile = Files.readAllLines(Paths.get(keyPath));
332                    keyList.addAll(keysFromFile);
333                } catch (IOException e) {
334                    System.err.println("Error reading alwayssendto file: " + e.getMessage());
335                }
336            }
337
338            forwardingKeys = keyList.stream()
339                                    .map(Base64.getDecoder()::decode)
340                                    .map(Key::new)
341                                    .collect(Collectors.toList());
342        } else {
343            forwardingKeys = alwaysSendToKeys;
344        }
345
346        return new Config(jdbcConfig, serverConfig, peerList, keyData, forwardingKeys, unixSocketFile, useWhiteList);
347    }
