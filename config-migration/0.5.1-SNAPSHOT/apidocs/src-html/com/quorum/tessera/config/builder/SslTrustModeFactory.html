002package com.quorum.tessera.config.builder;
003
004import com.quorum.tessera.config.SslTrustMode;
005import java.util.Collections;
006import java.util.HashMap;
007import java.util.Map;
008
009
010public interface SslTrustModeFactory {
011    
012     Map<String, SslTrustMode> TRUST_MODE_LOOKUP = 
013             Collections.unmodifiableMap(new HashMap<String, SslTrustMode>() {
014        {
015            put("ca", SslTrustMode.CA);
016            put("tofu", SslTrustMode.TOFU);
017            put("ca-or-tofu", SslTrustMode.CA_OR_TOFU);
018            put("whitelist", SslTrustMode.WHITELIST);
019            put("insecure-no-validation",SslTrustMode.NONE);
020            put("none", SslTrustMode.NONE);
021        }
022    });
023    
024    static SslTrustMode resolveByLegacyValue(String value) {
025        return TRUST_MODE_LOOKUP.getOrDefault(value, SslTrustMode.NONE);
026    }
027    
028    
029}
