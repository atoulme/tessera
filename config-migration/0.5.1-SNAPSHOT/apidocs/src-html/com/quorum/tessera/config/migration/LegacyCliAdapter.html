025public class LegacyCliAdapter implements CliAdapter {
026
027    private final FilesDelegate fileDelegate = FilesDelegate.create();
028
029    private final ConfigFactory configFactory;
030
031    public LegacyCliAdapter() {
032        this(new TomlConfigFactory());
033    }
034
035    protected LegacyCliAdapter(ConfigFactory configFactory) {
036        this.configFactory = Objects.requireNonNull(configFactory);
037    }
038
039    @Override
040    public CliResult execute(String... args) throws Exception {
041
042        Options options = buildOptions();
043        final List<String> argsList = Arrays.asList(args);
044        if (argsList.isEmpty() || argsList.contains("help")) {
045            String header = "Generate Tessera JSON config file from a Constellation TOML config file";
046            HelpFormatter formatter = new HelpFormatter();
047            formatter.printHelp("tessera-config-migration", header, options, null);
048            final int exitCode = argsList.isEmpty() ? 1 : 0;
049            return new CliResult(exitCode, true, false, null);
050        }
051
052        CommandLineParser parser = new DefaultParser();
053
054        CommandLine line = parser.parse(options, args);
055
056        final ConfigBuilder configBuilder = Optional.ofNullable(line.getOptionValue("tomlfile"))
057                                                .map(Paths::get)
058                                                .map(fileDelegate::newInputStream)
059                                                .map(stream -> this.configFactory.create(stream, null))
060                                                .map(ConfigBuilder::from)
061                                                .orElse(ConfigBuilder.create());
062
063        ConfigBuilder adjustedConfig = applyOverrides(line, configBuilder);
064
065        Config config = adjustedConfig.build();
066
067        String outputfilevalue = line.getOptionValue("outputfile", "tessera-config.json");
068
069        Path outputPath = Paths.get(outputfilevalue).toAbsolutePath();
070
071        return writeToOutputFile(config, outputPath);
072    }
073
074    static CliResult writeToOutputFile(Config config, Path outputPath) throws IOException {
075
076        System.out.printf("Saving config to  %s", outputPath);
077        System.out.println();
078        System.out.println(JaxbUtil.marshalToStringNoValidation(config));
079        System.out.println();
080
081        try (OutputStream outputStream = Files.newOutputStream(outputPath)) {
082            JaxbUtil.marshal(config, outputStream);
083            System.out.printf("Saved config to  %s", outputPath);
084            System.out.println();
085            return new CliResult(0, false, false, config);
086        } catch (ConstraintViolationException validationException) {
087            validationException.getConstraintViolations().stream()
088                    .forEach(System.err::println);
089            return new CliResult(2, false, false, config);
090        }
091    }

093    static Optional<Path> resolveUnixFilePath(Path initial, String workdir, String fileName) {
094        if (Objects.nonNull(workdir) && Objects.nonNull(fileName)) {
095            return Optional.of(Paths.get(workdir, fileName));
096        } else if(Objects.nonNull(fileName)) {
097            return Optional.of(Paths.get(fileName));
098        }
099
100        return Optional.ofNullable(initial);
101
102    }
103
104    static Optional<List<Path>> resolveListOfUnixFilePaths(List<Path> initial, String workdir, List<String> fileNames) {
105        if(Objects.nonNull(workdir) && Objects.nonNull(fileNames)) {
106            List<Path> paths = new ArrayList<>();
107            for(String fileName : fileNames) {
108                paths.add(Paths.get(workdir, fileName));
109            }
110            return Optional.of(paths);
111        }
112
113        return Optional.ofNullable(initial);
114    }

116    static ConfigBuilder applyOverrides(CommandLine line, ConfigBuilder configBuilder) {
117
118        Config initialConfig = configBuilder.build();
119
120        Optional.ofNullable(line.getOptionValue("url"))
121                .ifPresent(configBuilder::serverHostname);
122
123        Optional.ofNullable(line.getOptionValue("port"))
124                .map(Integer::valueOf)
125                .ifPresent(configBuilder::serverPort);
126
127        resolveUnixFilePath(initialConfig.getUnixSocketFile(), line.getOptionValue("workdir"), line.getOptionValue("socket"))
128                .ifPresent(configBuilder::unixSocketFile);
129
130        Optional.ofNullable(line.getOptionValues("othernodes"))
131                .map(Arrays::asList)
132                .ifPresent(configBuilder::peers);
133
134        KeyDataBuilder keyDataBuilder = KeyDataBuilder.create();
135
136        Optional.ofNullable(line.getOptionValues("publickeys"))
137                .map(Arrays::asList)
138                .ifPresent(keyDataBuilder::withPublicKeys);
139
140        Optional.ofNullable(line.getOptionValues("privatekeys"))
141                .map(Arrays::asList)
142                .ifPresent(keyDataBuilder::withPrivateKeys);
143
144        Optional.ofNullable(line.getOptionValues("alwayssendto"))
145                .map(Arrays::asList)
146                .ifPresent(configBuilder::alwaysSendTo);
147
148        Optional.ofNullable(line.getOptionValue("workdir"))
149                .ifPresent(keyDataBuilder::withWorkingDirectory);
150
151        resolveUnixFilePath(initialConfig.getKeys().getPasswordFile(), line.getOptionValue("workdir"), line.getOptionValue("passwords"))
152            .ifPresent(keyDataBuilder::withPrivateKeyPasswordFile);
153
154        Optional.ofNullable(line.getOptionValue("storage"))
155                .map(JdbcConfigFactory::fromLegacyStorageString)
156                .ifPresent(configBuilder::jdbcConfig);
157
158        Optional.ofNullable(line.getOptionValue("ipwhitelist"))
159                .ifPresent(v -> configBuilder.useWhiteList(true));
160
161        Optional.ofNullable(line.getOptionValue("tls"))
162                .map(String::toUpperCase)
163                .map(SslAuthenticationMode::valueOf)
164                .ifPresent(configBuilder::sslAuthenticationMode);
165
166        Optional.ofNullable(line.getOptionValue("tlsservertrust"))
167                .map(SslTrustModeFactory::resolveByLegacyValue)
168                .ifPresent(configBuilder::sslServerTrustMode);
169
170        Optional.ofNullable(line.getOptionValue("tlsclienttrust"))
171                .map(SslTrustModeFactory::resolveByLegacyValue)
172                .ifPresent(configBuilder::sslClientTrustMode);
173
174        resolveUnixFilePath(initialConfig.getServerConfig().getSslConfig().getServerTlsCertificatePath(), line.getOptionValue("workdir"), line.getOptionValue("tlsservercert"))
175            .ifPresent(configBuilder::sslServerTlsCertificatePath);
176
177        resolveUnixFilePath(initialConfig.getServerConfig().getSslConfig().getClientTlsCertificatePath(), line.getOptionValue("workdir"), line.getOptionValue("tlsclientcert"))
178            .ifPresent(configBuilder::sslClientTlsCertificatePath);
179
180        Optional.ofNullable(line.getOptionValues("tlsserverchain"))
181                .map(Arrays::asList)
182                .ifPresent(l -> {
183                    List<Path> sslServerTrustCertificates = resolveListOfUnixFilePaths(
184                        initialConfig.getServerConfig().getSslConfig().getServerTrustCertificates(),
185                        line.getOptionValue("workdir"),
186                        l
187                    ).get();
188                    configBuilder.sslServerTrustCertificates(sslServerTrustCertificates);
189                });
190
191        Optional.ofNullable(line.getOptionValues("tlsclientchain"))
192                .map(Arrays::asList)
193                .ifPresent(l -> {
194                    List<Path> sslClientTrustCertificates = resolveListOfUnixFilePaths(
195                        initialConfig.getServerConfig().getSslConfig().getClientTrustCertificates(),
196                        line.getOptionValue("workdir"),
197                        l
198                    ).get();
199                    configBuilder.sslClientTrustCertificates(sslClientTrustCertificates);
200                });
201
202        resolveUnixFilePath(initialConfig.getServerConfig().getSslConfig().getServerTlsKeyPath(), line.getOptionValue("workdir"), line.getOptionValue("tlsserverkey"))
203            .ifPresent(configBuilder::sslServerTlsKeyPath);
204
205        resolveUnixFilePath(initialConfig.getServerConfig().getSslConfig().getClientTlsKeyPath(), line.getOptionValue("workdir"), line.getOptionValue("tlsclientkey"))
206            .ifPresent(configBuilder::sslClientTlsKeyPath);
207
208        resolveUnixFilePath(initialConfig.getServerConfig().getSslConfig().getKnownServersFile(), line.getOptionValue("workdir"), line.getOptionValue("tlsknownservers"))
209            .ifPresent(configBuilder::sslKnownServersFile);
210
211        resolveUnixFilePath(initialConfig.getServerConfig().getSslConfig().getKnownClientsFile(), line.getOptionValue("workdir"), line.getOptionValue("tlsknownclients"))
212            .ifPresent(configBuilder::sslKnownClientsFile);
213
214        final KeyConfiguration keyConfiguration = keyDataBuilder.build();
215
216        if (!keyConfiguration.getKeyData().isEmpty()) {
217            configBuilder.keyData(keyConfiguration);
218        }
219
220        return configBuilder;
221    }
