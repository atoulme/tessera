032public class TomlConfigFactory implements ConfigFactory {
033
034    private static final Logger LOGGER = LoggerFactory.getLogger(TomlConfigFactory.class);
035
036    private final FilesDelegate filesDelegate;
037
038    public TomlConfigFactory() {
039        this(FilesDelegate.create());
040    }
041
042    public TomlConfigFactory(FilesDelegate filesDelegate) {
043        this.filesDelegate = Objects.requireNonNull(filesDelegate);
044    }
045
046    @Override
047    public Config create(InputStream configData, ArgonOptions options, String... filenames) {
048        Objects.requireNonNull(configData, "No config data provided. ");
049        if (filenames.length != 0) {
050            throw new UnsupportedOperationException("keyConfigData arg is not implemented for TomlConfigFactory");
051        }
052
053        Toml toml = new Toml().read(configData);
054        toml.toMap().entrySet().stream().forEach(entry -> {
055            LOGGER.debug("Found entry in toml file : {} {}", entry.getKey(), entry.getValue());
056        });
057
058        final String url = toml.getString("url");
059
060        final Integer port = Optional.ofNullable(toml.getLong("port"))
061                                                    .map(Long::intValue)
062                                                    .orElse(null);
063
064        final String workdir = toml.getString("workdir", "");
065        final String socket = toml.getString("socket");
066
067        final Path unixSocketFile;
068        if(socket != null) {
069            unixSocketFile = Paths.get(workdir, socket);
070        } else {
071            unixSocketFile = null;
072        }
073
074        final String tls = toml.getString("tls", "strict").toUpperCase();
075
076        final List<String> othernodes = toml.getList("othernodes", Collections.emptyList());
077
078        final List<String> publicKeyList = toml.getList("publickeys", Collections.emptyList());
079
080        final List<String> privateKeyList = toml.getList("privatekeys", Collections.emptyList());
081
082        final Optional<String> privateKeyPasswordFile = Optional.ofNullable(toml.getString("passwords"));
083        final Path privateKeyPasswordPath;
084        if(privateKeyPasswordFile.isPresent()) {
085            privateKeyPasswordPath = Paths.get(workdir, privateKeyPasswordFile.get());
086        } else {
087            privateKeyPasswordPath = null;
088        }
089
090        KeyConfiguration keyData;
091        if(!publicKeyList.isEmpty() || !privateKeyList.isEmpty()) {
092            keyData = KeyDataBuilder.create()
093                                    .withPublicKeys(publicKeyList)
094                                    .withPrivateKeys(privateKeyList)
095                                    .withPrivateKeyPasswordFile(privateKeyPasswordPath)
096                                    .withWorkingDirectory(workdir)
097                                    .build();
098        } else {
099            keyData = new KeyConfiguration(null, null, null);
100        }
101
102        final List<String> alwaysSendToKeyPaths = toml.getList("alwayssendto", Collections.emptyList());
103
104        final String storage = toml.getString("storage", "memory");
105
106        final List<String> ipwhitelist = toml.getList("ipwhitelist", Collections.EMPTY_LIST);
107        final boolean useWhiteList = !ipwhitelist.isEmpty();
108
109        //Server side
110        final String tlsservertrust = toml.getString("tlsservertrust", "tofu");
111
112        final Optional<String> tlsserverkeyStr = Optional.ofNullable(toml.getString("tlsserverkey"));
113        final Path tlsserverkey = tlsserverkeyStr.map(s -> Paths.get(workdir, s)).orElse(null);
114
115        final Optional<String> tlsservercertStr = Optional.ofNullable(toml.getString("tlsservercert"));
116        final Path tlsservercert = tlsservercertStr.map(s -> Paths.get(workdir, s)).orElse(null);
117
118        final List<String> tlsserverchainnames = toml.getList("tlsserverchain", Collections.emptyList());
119        List<Path> tlsserverchain = new ArrayList<>();
120        for(String name : tlsserverchainnames) {
121            tlsserverchain.add(Paths.get(workdir, name));
122        }
123
124        final Optional<String> tlsknownclientsStr = Optional.ofNullable(toml.getString("tlsknownclients"));
125        final Path tlsknownclients = tlsknownclientsStr.map(s -> Paths.get(workdir, s)).orElse(null);
126
127        //Client side
128        final String tlsclienttrust = toml.getString("tlsclienttrust", "tofu");
129
130        final Optional<String> tlsclientkeyStr = Optional.ofNullable(toml.getString("tlsclientkey"));
131        final Path tlsclientkey = tlsclientkeyStr.map(s -> Paths.get(workdir, s)).orElse(null);
132
133        final Optional<String> tlsclientcertStr = Optional.ofNullable(toml.getString("tlsclientcert"));
134        final Path tlsclientcert = tlsclientcertStr.map(s -> Paths.get(workdir, s)).orElse(null);
135
136        final List<String> tlsclientchainnames = toml.getList("tlsclientchain", Collections.emptyList());
137        List<Path> tlsclientchain = new ArrayList<>();
138        for(String name : tlsclientchainnames) {
139            tlsclientchain.add(Paths.get(workdir, name));
140        }
141
142        final Optional<String> tlsknownserversStr = Optional.ofNullable(toml.getString("tlsknownservers"));
143        final Path tlsknownservers = tlsknownserversStr.map(s -> Paths.get(workdir, s)).orElse(null);
144
145        ConfigBuilder configBuilder = ConfigBuilder.create()
146                .serverPort(port)
147                .serverHostname(url)
148                .unixSocketFile(unixSocketFile)
149                .sslAuthenticationMode(SslAuthenticationMode.valueOf(tls))
150                .sslServerTrustMode(SslTrustModeFactory.resolveByLegacyValue(tlsservertrust))
151                .sslServerTlsKeyPath(tlsserverkey)
152                .sslServerTlsCertificatePath(tlsservercert)
153                .sslServerTrustCertificates(tlsserverchain)
154                .sslKnownClientsFile(tlsknownclients)
155                .sslClientTrustMode(SslTrustModeFactory.resolveByLegacyValue(tlsclienttrust))
156                .sslClientTlsKeyPath(tlsclientkey)
157                .sslClientTlsCertificatePath(tlsclientcert)
158                .sslClientTrustCertificates(tlsclientchain)
159                .sslKnownServersFile(tlsknownservers)
160                .peers(othernodes)
161                .alwaysSendTo(alwaysSendToKeyPaths)
162                .useWhiteList(useWhiteList)
163                .keyData(keyData);
164
165        Optional.ofNullable(storage)
166                .map(JdbcConfigFactory::fromLegacyStorageString).ifPresent(configBuilder::jdbcConfig);
167
168        return configBuilder.build();
169    }

171    static List<KeyDataConfig> createPrivateKeyData(List<String> privateKeys, List<String> privateKeyPasswords) {
172
173        //Populate null values assume that they arent private
174        List<String> passwordList = new ArrayList<>(privateKeyPasswords);
175        for (int i = privateKeyPasswords.size() - 1; i < privateKeys.size(); i++) {
176            passwordList.add(null);
177        }
178
179        List<JsonObject> privateKeyJson = privateKeys
180                .stream()
181                .map(Paths::get)
182                .map(path -> IOCallback.execute(() -> Files.newInputStream(path)))
183                .map(Json::createReader)
184                .map(JsonReader::readObject)
185                .collect(Collectors.toList());
186
187        List<KeyDataConfig> privateKeyData = IntStream
188                .range(0, privateKeyJson.size())
189                //FIXME: Canyt set to null value.. need to use addNull("password")
190                .mapToObj(i -> {
191
192                    final String password = passwordList.get(i);
193                    final JsonObject keyDatC = Json.createObjectBuilder(privateKeyJson.get(i)).build();
194
195                    final JsonObject dataNode = keyDatC.getJsonObject("data");
196                    final JsonObjectBuilder ammendedDataNode = Json.createObjectBuilder(dataNode);
197
198                    boolean isLocked = Objects.equals(keyDatC.getString("type"), "argon2sbox");
199                    if (isLocked) {
200                        ammendedDataNode.add("password", Objects.requireNonNull(password, "Password is required."));
201                    }
202
203                    return Json.createObjectBuilder(keyDatC)
204                            .remove("data")
205                            .add("data", ammendedDataNode)
206                            .build();
207                })
208                .map(JsonObject::toString)
209                .map(String::getBytes)
210                .map(ByteArrayInputStream::new)
211                .map(inputStream -> JaxbUtil.unmarshal(inputStream, KeyDataConfig.class))
212                .collect(Collectors.toList());
213
214        return Collections.unmodifiableList(privateKeyData);
215    }
